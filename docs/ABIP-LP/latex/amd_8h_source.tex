\DoxyCodeLine{\Hypertarget{amd_8h_source_l00011}00011 \textcolor{comment}{/* AMD finds a symmetric ordering P of a matrix A so that the Cholesky}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00012}00012 \textcolor{comment}{ * factorization of P*A*P' has fewer nonzeros and takes less work than the}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00013}00013 \textcolor{comment}{ * Cholesky factorization of A.  If A is not symmetric, then it performs its}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00014}00014 \textcolor{comment}{ * ordering on the matrix A+A'.  Two sets of user-\/callable routines are}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00015}00015 \textcolor{comment}{ * provided, one for int integers and the other for SuiteSparse\_long integers.}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00016}00016 \textcolor{comment}{ *}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00017}00017 \textcolor{comment}{ * The method is based on the approximate minimum degree algorithm, discussed}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00018}00018 \textcolor{comment}{ * in Amestoy, Davis, and Duff, "{}An approximate degree ordering algorithm"{},}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00019}00019 \textcolor{comment}{ * SIAM Journal of Matrix Analysis and Applications, vol. 17, no. 4, pp.}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00020}00020 \textcolor{comment}{ * 886-\/905, 1996.  This package can perform both the AMD ordering (with}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00021}00021 \textcolor{comment}{ * aggressive absorption), and the AMDBAR ordering (without aggressive}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00022}00022 \textcolor{comment}{ * absorption) discussed in the above paper.  This package differs from the}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00023}00023 \textcolor{comment}{ * Fortran codes discussed in the paper:}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00024}00024 \textcolor{comment}{ *}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00025}00025 \textcolor{comment}{ *       (1) it can ignore "{}dense"{} rows and columns, leading to faster run times}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00026}00026 \textcolor{comment}{ *       (2) it computes the ordering of A+A' if A is not symmetric}}
\DoxyCodeLine{\Hypertarget{amd_8h_source_l00027}00027 \textcolor{comment}{ *       (3) it is followed by a depth-\/first post-\/ordering of the assembly tree}}

\DoxyCodeLine{\Hypertarget{amd_8h_source_l00029}00029 \textcolor{comment}{ *}}
