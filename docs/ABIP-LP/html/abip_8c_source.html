458static void calc_residuals
459 (
460 ABIPWork *w,
461 ABIPResiduals *r,
462 abip_int ipm_iter,
463 abip_int admm_iter
464 )
465{
466 DEBUG_FUNC
467
468 abip_float *y;
469 abip_float *x;
470 abip_float *s;
471
472 // find y, x, s
473 if (w->stgs->avg_criterion)
474 {
475 y = w->u_avgcon;
476 x = &(w->u_avgcon[w->m]);
477 s = &(w->v_avgcon[w->m]);
478 }
479 else
480 {
481 y = w->u;
482 x = &(w->u[w->m]);
483 s = &(w->v[w->m]);
484 }
485
486 abip_float nmpr_tau;
487 abip_float nmdr_tau;
488 abip_float nm_A_x_tau;
489 abip_float nm_At_ys_tau;
490 abip_float ct_x;
491 abip_float bt_y;
492
493 abip_int n = w->n;
494 abip_int m = w->m;
495
496 if (admm_iter && r->last_admm_iter == admm_iter)
497 {
498 RETURN;
499 }
500
501 r->last_ipm_iter = ipm_iter;
502 r->last_admm_iter = admm_iter;
503
504
505 if (w->stgs->avg_criterion)
506 {
507 r->tau = ABS(w->u_avgcon[n + m]);
508 r->kap = ABS(w->v_avgcon[n + m]) / (w->stgs->normalize ? (w->stgs->scale * w->sc_c * w->sc_b) : 1);
509 }
510 else
511 {
512 r->tau = ABS(w->u[n + m]);
513 r->kap = ABS(w->v[n + m]) / (w->stgs->normalize ? (w->stgs->scale * w->sc_c * w->sc_b) : 1);
514 }
515
516
517 // compute primal resid. and dual resid. memset pr and dr, then compute.
518 nmpr_tau = calc_primal_resid(w, x, r->tau, &nm_A_x_tau);
519 nmdr_tau = calc_dual_resid(w, y, s, r->tau, &nm_At_ys_tau);
520
521 r->bt_y_by_tau = ABIP(dot)(y, w->b, m) / (w->stgs->normalize ? (w->stgs->scale * w->sc_c * w->sc_b) : 1);
522 r->ct_x_by_tau = ABIP(dot)(x, w->c, n) / (w->stgs->normalize ? (w->stgs->scale * w->sc_c * w->sc_b) : 1);
523
524 r->res_infeas = r->bt_y_by_tau > 0 ? w->nm_b * nm_At_ys_tau / r->bt_y_by_tau : NAN;
525 r->res_unbdd = r->ct_x_by_tau < 0 ? w->nm_c * nm_A_x_tau / -r->ct_x_by_tau : NAN;
526
527 bt_y = SAFEDIV_POS(r->bt_y_by_tau, r->tau);
528 ct_x = SAFEDIV_POS(r->ct_x_by_tau, r->tau);
529
530 r->res_pri = SAFEDIV_POS(nmpr_tau / (1 + w->nm_b), r->tau);
531 r->res_dual = SAFEDIV_POS(nmdr_tau / (1 + w->nm_c), r->tau);
532 r->rel_gap = ABS(ct_x - bt_y) / (1 + ABS(ct_x) + ABS(bt_y));
533
534 RETURN;
535}

857 else if (ratio > 0.005 && ratio <= 0.01)
858 {
859 gamma = 0.5;
860 }
861 else if (ratio > 0.001 && ratio <= 0.005)
862 {
863 gamma = 0.4;
864 }
865 else
866 {
867 gamma = 0.3;
868 }
869
870 if (err_ratio > 6 && err_ratio <= 10)
871 {
872 sigma = 0.82;
873 gamma = gamma * 0.8;
874 }
875 else if (err_ratio > 4 && err_ratio <= 6)
876 {
877 sigma = 0.84;
878 gamma = gamma * 0.6;
879 }
880 else if (err_ratio > 3 && err_ratio <= 4)
881 {
882 sigma = 0.85;
883 gamma = gamma * 0.5;
884 w->final_check = 1;
885 }
886 else if (err_ratio > 1 && err_ratio <= 3)
887 {
888 w->final_check = 1;
889 if (ratio < 0.1)
890 {
891 if (w->double_check)
892 {
893 sigma = 0.9;
894 gamma = gamma * 0.4;
895 w->double_check = 0;
896 }
897 else
898 {
899 sigma = 1.0;
900 gamma = gamma * 0.1;
901 w->double_check = 1;
902 }
903 }
904 else
905 {
906 sigma = 0.88;
907 gamma = gamma * 0.4;
908 }
909 }
910 else
911 {
912 sigma = w->sigma;
913 }
914 }
915
916 mu = mu * sigma;
917
918 w->mu = mu;
919 w->sigma = sigma;
920 w->gamma = gamma;
921}

930static void update_barrier_dynamic
931 (
932 ABIPWork *w,
933 ABIPResiduals *r
934 )
935{
936 /*
937 Implementation of a more delicate mu adjustment strategy
938
939 ksi = min_i {x_i * z_i} / (x' * z / n)
940 sigma = 0.1 * min(0.05 * (1 - ksi) / ksi, 2)^3
941
942 */
943
944 abip_int m = w->m, n = w->n, l = m + n + 1, i;
945
946 double *u;
947 double *v;
948 double shrink = w->stgs->dynamic_sigma;
949
950 if (w->stgs->avg_criterion){
951 u = w->u_avgcon;
952 v = w->v_avgcon;
953 }
954 else
955 {
956 u = w->u;
957 v = w->v;
958 }
959
960 double ksi = 0.0, sigma = 0.0, xisi = 0.0, xs = 0.0, minxs = 1e+10;
961
962 for (i = m; i < l; ++i) {
963 xisi = u[i] * v[i];
964 xs += xisi; minxs = MIN(xisi, minxs);
965 }
966
967 if (minxs <= 0.0) {
968 abip_printf("Invalid xisi < 0 \n");
969 assert(0);
970 }
971
972 xs /= (n + 1); ksi = minxs / xs;
973 sigma = MIN(0.05 * (1 - ksi) / ksi, 2.0);
974 sigma = MAX(0.1 * sigma * sigma * sigma, shrink);
975 w->mu *= sigma;
976 // w->gamma = gamma;
977}

1951static abip_float iterate_Q_norm_resd
1952 (
1953 ABIPWork *w,
1954 abip_int j
1955 )
1956{
1957 DEBUG_FUNC
1958
1959 abip_int i;
1960 abip_int l = w->m + w->n + 1;
1961
1962 abip_float *y = w->u;
1963 abip_float *x = &(w->u[w->m]);
1964 abip_float *s = &(w->v[w->m]);
1965 abip_float tau = w->u[w->m + w->n];
1966 abip_float kap = w->v[w->m + w->n];
1967
1968 abip_float Qres = 0;
1969 abip_float Qres_avg = w->stgs->max_admm_iters;
1970 abip_float norm_avg = 1;
1971
1972 // initialize pr and dr
1973 abip_float *pr = w->pr;
1974 abip_float *dr = w->dr;
1975
1976 memset(pr, 0, w->m * sizeof(abip_float));
1977 memset(dr, 0, w->n * sizeof(abip_float));
1978
1979 // compute pr and dr
1980 ABIP(accum_by_A)(w->A, w->p, x, pr);
1981 ABIP(accum_by_Atrans)(w->A, w->p, y, dr);
1982 ABIP(add_scaled_array)(dr, s, w->n, 1.0);
1983
1984 for (i = 0; i < w->m; ++i)
1985 {
1986 Qres += (pr[i] - w->b[i] * tau) * (pr[i] - w->b[i] * tau);
1987 }
1988
1989 for (i = 0; i < w->n; ++i)
1990 {
1991 Qres += (dr[i] - w->c[i] * tau) * (dr[i] - w->c[i] * tau);
1992 }
1993 abip_float cTx = ABIP(dot)(x, w->c, w->n);
1994 abip_float bTy = ABIP(dot)(y, w->b, w->m);
1995 Qres += (bTy - cTx - kap) * (bTy - cTx - kap);
1996 abip_float norm = 1 + SQRTF(ABIP(norm_sq)(w->u, l) + ABIP(norm_sq)(w->v, l)) ;
1997
1998
1999 // check inner loop termination criterion via the average solution returned by restart strategy
2000 if ( (j+1) % 10 == 0)
2001 {
2002 // initialize pr_avg and dr_avg
2003 Qres_avg = 0;
2004 norm_avg = 0;
2005 abip_float *y_avg = w->u_avgcon;
2006 abip_float *x_avg = &(w->u_avgcon[w->m]);
2007 abip_float *s_avg = &(w->v_avgcon[w->m]);
2008 abip_float tau_avg = w->u_avgcon[w->m + w->n];
2009 abip_float kap_avg = w->v_avgcon[w->m + w->n];
2010 abip_float *pr_avg;
2011 abip_float *dr_avg;
2012 pr_avg = (abip_float *) abip_malloc(w->m * sizeof(abip_float));
2013 dr_avg = (abip_float *) abip_malloc(w->n * sizeof(abip_float));
2014 memset(pr_avg, 0, w->m * sizeof(abip_float));
2015 memset(dr_avg, 0, w->n * sizeof(abip_float));
2016
2017 // compute pr_avg and dr_avg
2018 ABIP(accum_by_A)(w->A, w->p, x_avg, pr_avg);
2019 ABIP(accum_by_Atrans)(w->A, w->p, y_avg, dr_avg);
2020 ABIP(add_scaled_array)(dr_avg, s_avg, w->n, 1.0);
2021 for (i = 0; i < w->m; ++i)
2022 {
2023 Qres_avg += (pr_avg[i] - w->b[i] * tau_avg) * (pr_avg[i] - w->b[i] * tau_avg);
2024 }
2025
2026 for (i = 0; i < w->n; ++i)
2027 {
2028 Qres_avg += (dr_avg[i] - w->c[i] * tau_avg) * (dr_avg[i] - w->c[i] * tau_avg);
2029 }
2030
2031 abip_float cTx_avg = ABIP(dot)(x_avg, w->c, w->n);
2032 abip_float bTy_avg = ABIP(dot)(y_avg, w->b, w->m);
2033
2034 Qres_avg += (bTy_avg - cTx_avg - kap_avg) * (bTy_avg - cTx_avg - kap_avg);
2035
2036 norm_avg = 1 + SQRTF(ABIP(norm_sq)(w->u_avgcon, l) + ABIP(norm_sq)(w->v_avgcon, l));
2037 abip_free(pr_avg); abip_free(dr_avg);
2038 }
2039
2040 if (SQRTF(Qres_avg) / norm_avg < SQRTF(Qres) / norm )
2041 {
2042 w->stgs->avg_criterion = 1;
2043
2044 RETURN SQRTF(Qres_avg) / norm_avg;
2045 }
2046 else
2047 {
2048 w->stgs->avg_criterion = 0;
2049 RETURN SQRTF(Qres) / norm;
2050 }
2051}
