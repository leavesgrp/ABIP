104
105#if EXTRA_VERBOSE > 0
106 ABIP(timer) transpose_timer;
107 abip_printf("transposing A\n");
108 ABIP(tic)(&transpose_timer);
109#endif
110
111 z = (abip_int *)abip_calloc(m, sizeof(abip_int));
112
113 for (i = 0; i < Ap[n]; i++)
114 {
115 z[Ai[i]]++; /* row counts */
116 }
117
118 ABIP(cumsum)(Cp, z, m); /* row pointers */
119
120 for (j = 0; j < n; j++)
121 {
122 c1 = Ap[j];
123 c2 = Ap[j + 1];
124 for (i = c1; i < c2; i++)
125 {
126 q = z[Ai[i]];
127 Ci[q] = j; /* place A(i,j) as entry C(j,i) */
128 Cx[q] = Ax[i];
129 z[Ai[i]]++;
130 }
131 }
132
133 abip_free(z);
134
135#if EXTRA_VERBOSE > 0

320/* solves (I+AA')x = b, s warm start, solution stored in b */
321static abip_int pcg
322 (
323 const ABIPMatrix *A,
324 const ABIPSettings *stgs,
325 ABIPLinSysWork *pr,
326 const abip_float *s,
327 abip_float *b,
328 abip_int max_its,
329 abip_float tol
330 )
331{
332 abip_int i;
333 abip_int m = A->m;
334
335 abip_float ipzr;
336 abip_float ipzr_old;
337 abip_float alpha;
338
339 abip_float *p = pr->p; /* cg direction */
340 abip_float *Gp = pr->Gp; /* updated CG direction */
341 abip_float *r = pr->r; /* cg residual */
342 abip_float *z = pr->z; /* for preconditioning */
343 abip_float *M = pr->M; /* inverse diagonal preconditioner */
344
345 if (s == ABIP_NULL)
346 {
347 memcpy(r, b, m * sizeof(abip_float));
348 memset(b, 0, m * sizeof(abip_float));
349 }
350 else
351 {
352 mat_vec(A, stgs, pr, s, r);
353 ABIP(add_scaled_array)(r, b, m, -1);
354 ABIP(scale_array)(r, -1, m);
355 memcpy(b, s, m * sizeof(abip_float));
356 }
357
358 /* check to see if we need to run CG at all */
359 if (ABIP(norm)(r, m) < MIN(tol, 1e-18))
360 {
361 return 0;
362 }
363
364 apply_pre_conditioner(M, z, r, m, &ipzr);
365 memcpy(p, z, m * sizeof(abip_float));
366
367 /* main loop */
368 for (i = 0; i < max_its; ++i)
369 {
370 mat_vec(A, stgs, pr, p, Gp);
371 alpha = ipzr / ABIP(dot)(p, Gp, m);
372 ABIP(add_scaled_array)(b, p, m, alpha);
373 ABIP(add_scaled_array)(r, Gp, m, -alpha);
374
375 if (ABIP(norm)(r, m) < tol)
376 {
377#if EXTRA_VERBOSE > 0
378 abip_printf("tol: %.4e, resid: %.4e, iters: %li\n", tol, ABIP(norm)(r, m), (long)i + 1);
379#endif
380
381 return i + 1;
382 }
383
384 ipzr_old = ipzr;
385 apply_pre_conditioner(M, z, r, m, &ipzr);
386 ABIP(scale_array)(p, ipzr / ipzr_old, m);
387 ABIP(add_scaled_array)(p, z, m, 1);
388 }
389
390 return i;
391}

393abip_int ABIP(solve_lin_sys)
394(
395 const ABIPMatrix *A,
396 const ABIPSettings *stgs,
397 ABIPLinSysWork *p,
398 abip_float *b,
399 const abip_float *s,
400 abip_int iter
401 )
402{

404 ABIP(timer) linsys_timer;
405
406 abip_float cg_tol = ABIP(norm)(b, A->m) * (iter < 0 ? CG_BEST_TOL
407 : CG_MIN_TOL / POWF((abip_float)iter + 1, stgs->cg_rate));
408
409 cg_tol = MAX(cg_tol, 1e-07);
410
411 ABIP(tic)(&linsys_timer);

413 /* solves Mx = b, for x but stores result in b */
414 /* s contains warm-start (if available) */
415 ABIP(accum_by_A)(A, p, &(b[A->m]), b);
416
417 /* solves (I+AA')x = b, s warm start, solution stored in b */
418 cg_its = pcg(A, stgs, p, s, b, A->m, MAX(cg_tol, CG_BEST_TOL));

420 ABIP(accum_by_Atrans)(A, p, b, &(b[A->m]));
421
422 if (iter >= 0)

425 }
426
