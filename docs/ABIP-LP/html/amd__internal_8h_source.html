101/* FLIP is a "negation about -1", and is used to mark an integer i that is
102 * normally non-negative. FLIP (EMPTY) is EMPTY. FLIP of a number > EMPTY
103 * is negative, and FLIP of a number < EMTPY is positive. FLIP (FLIP (i)) = i
104 * for all integers i. UNFLIP (i) is >= EMPTY. */
105#define EMPTY (-1)
106#define FLIP(i) (-(i)-2)
107#define UNFLIP(i) ((i < EMPTY) ? FLIP (i) : (i))
108
109/* for integer MAX/MIN, or for doubles when we don't care how NaN's behave: */
110#define MAX(a,b) (((a) > (b)) ? (a) : (b))
111#define MIN(a,b) (((a) < (b)) ? (a) : (b))
