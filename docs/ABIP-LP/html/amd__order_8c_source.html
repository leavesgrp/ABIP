21GLOBAL Int AMD_order
22(
23 Int n,
24 const Int Ap [ ],
25 const Int Ai [ ],
26 Int P [ ],
27 abip_float Control [ ],

48 abip_float mem = 0 ;
49
50 #ifndef NDEBUG
51 AMD_debug_init ("amd") ;
52 #endif
53
54 /* clear the ABIPInfo array, if it exists */
55 info = ABIPInfo != (abip_float *) ABIP_NULL ;
56
57 if (info)
58 {
59 for (i = 0 ; i < AMD_INFO ; i++)
60 {
61 ABIPInfo [i] = EMPTY ;
62 }
63 ABIPInfo [AMD_N] = n ;
64 ABIPInfo [AMD_STATUS] = AMD_OK ;
65 }
66
67 /* make sure inputs exist and n is >= 0 */
68 if (Ai == (Int *) ABIP_NULL || Ap == (Int *) ABIP_NULL || P == (Int *) ABIP_NULL || n < 0)
69 {
70 if (info) ABIPInfo [AMD_STATUS] = AMD_INVALID ;
71 return (AMD_INVALID) ; /* arguments are invalid */
72 }
73
74 if (n == 0)
75 {
76 return (AMD_OK) ; /* n is 0 so there's nothing to do */
77 }
78
79 nz = Ap [n] ;
80
81 if (info)
82 {
83 ABIPInfo [AMD_NZ] = nz ;
84 }
85
86 if (nz < 0)
87 {
88 if (info) ABIPInfo [AMD_STATUS] = AMD_INVALID ;
89 return (AMD_INVALID) ;
90 }
91
92 /* check if n or nz will cause size_t overflow */
93 if (((size_t) n) >= SIZE_T_MAX / sizeof (Int) || ((size_t) nz) >= SIZE_T_MAX / sizeof (Int))
94 {
95 if (info) ABIPInfo [AMD_STATUS] = AMD_OUT_OF_MEMORY ;
96 return (AMD_OUT_OF_MEMORY) ; /* problem too large */
97 }
98
99 /* check the input matrix: AMD_OK, AMD_INVALID, or AMD_OK_BUT_JUMBLED */
100 status = AMD_valid (n, n, Ap, Ai) ;
101
102 if (status == AMD_INVALID)
103 {
104 if (info) ABIPInfo [AMD_STATUS] = AMD_INVALID ;
105 return (AMD_INVALID) ; /* matrix is invalid */
106 }
107
108 /* allocate two size-n integer workspaces */
109 Len = amd_malloc (n * sizeof (Int)) ;
110 Pinv = amd_malloc (n * sizeof (Int)) ;
111 mem += n ;
112 mem += n ;
113
114 if (!Len || !Pinv)
115 {
116 /* :: out of memory :: */
117 amd_free (Len) ;
118 amd_free (Pinv) ;
119 if (info) ABIPInfo [AMD_STATUS] = AMD_OUT_OF_MEMORY ;
120 return (AMD_OUT_OF_MEMORY) ;
121 }
122
123 if (status == AMD_OK_BUT_JUMBLED)
124 {
125 /* sort the input matrix and remove duplicate entries */
126 AMD_DEBUG1 (("Matrix is jumbled\n")) ;
127 Rp = amd_malloc ((n+1) * sizeof (Int));
128 Ri = amd_malloc (nz * sizeof (Int));
129 mem += (n+1) ;
130 mem += MAX (nz,1) ;
131
132 if (!Rp || !Ri)
133 {
134 /* :: out of memory :: */
135 amd_free (Rp) ;
136 amd_free (Ri) ;
137 amd_free (Len) ;
138 amd_free (Pinv) ;
139
140 if (info) ABIPInfo [AMD_STATUS] = AMD_OUT_OF_MEMORY ;
141 return (AMD_OUT_OF_MEMORY) ;
142 }
143
144 /* use Len and Pinv as workspace to create R = A' */
145 AMD_preprocess (n, Ap, Ai, Rp, Ri, Len, Pinv) ;
146 Cp = Rp ;
147 Ci = Ri ;
148 }
149 else
150 {
151 /* order the input matrix as-is. No need to compute R = A' first */
152 Rp = ABIP_NULL ;
153 Ri = ABIP_NULL ;
154 Cp = (Int *) Ap ;
155 Ci = (Int *) Ai ;
156 }
157
158 /* --------------------------------------------------------------------- */
159 /* determine the symmetry and count off-diagonal nonzeros in A+A' */
160 /* --------------------------------------------------------------------- */
161
162 nzaat = AMD_aat (n, Cp, Ci, Len, P, ABIPInfo) ;
163 AMD_DEBUG1 (("nzaat: %g\n", (abip_float) nzaat)) ;
164 ASSERT ((MAX (nz-n, 0) <= nzaat) && (nzaat <= 2 * (size_t) nz)) ;
165
166 /* --------------------------------------------------------------------- */
167 /* allocate workspace for matrix, elbow room, and 6 size-n vectors */
168 /* --------------------------------------------------------------------- */
169
170 S = ABIP_NULL ;
171 slen = nzaat ; /* space for matrix */
172 ok = ((slen + nzaat/5) >= slen) ; /* check for size_t overflow */
173 slen += nzaat/5 ; /* add elbow room */
174
175 for (i = 0 ; ok && i < 7 ; i++)
176 {
177 ok = ((slen + n) > slen) ; /* check for size_t overflow */
178 slen += n ; /* size-n elbow room, 6 size-n work */
179 }
180
181 mem += slen ;
182 ok = ok && (slen < SIZE_T_MAX / sizeof (Int)) ; /* check for overflow */
183 ok = ok && (slen < Int_MAX) ; /* S[i] for Int i must be OK */
184
185 if (ok)
186 {
187 S = amd_malloc (slen * sizeof (Int)) ;
188 }
189
190 AMD_DEBUG1 (("slen %g\n", (abip_float) slen)) ;
191
192 if (!S)
193 {
194 /* :: out of memory :: (or problem too large) */
195 amd_free (Rp) ;
196 amd_free (Ri) ;
197 amd_free (Len) ;
198 amd_free (Pinv) ;
199 if (info) ABIPInfo [AMD_STATUS] = AMD_OUT_OF_MEMORY ;
200 return (AMD_OUT_OF_MEMORY) ;
201 }
202
203 if (info)
204 {
205 /* memory usage, in bytes. */
206 ABIPInfo [AMD_MEMORY] = mem * sizeof (Int) ;
207 }
208
209 /* --------------------------------------------------------------------- */
210 /* order the matrix */
211 /* --------------------------------------------------------------------- */
212
213 AMD_1 (n, Cp, Ci, P, Pinv, Len, slen, S, Control, ABIPInfo) ;
214
215 /* --------------------------------------------------------------------- */
216 /* free the workspace */
217 /* --------------------------------------------------------------------- */
218
219 amd_free (Rp) ;
220 amd_free (Ri) ;
221 amd_free (Len) ;
222 amd_free (Pinv) ;
223 amd_free (S) ;
224
225 if (info) ABIPInfo [AMD_STATUS] = status ;
226 return (status) ; /* successful ordering */
227}
