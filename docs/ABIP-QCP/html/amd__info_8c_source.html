11/* User-callable. Prints the output statistics for AMD. See amd.h
12 * for details. If the ABIPInfo array is not present, nothing is printed.
13 */
14
15#include "amd_internal.h"

17#define PRI(format,x) { if (x >= 0) { PRINTF ((format, x)) ; }}

19GLOBAL void AMD_info
20(
21 abip_float ABIPInfo [ ]
22)
23{
24 abip_float n;
25 abip_float ndiv;
26 abip_float nmultsubs_ldl;
27 abip_float nmultsubs_lu;
28 abip_float lnz;
29 abip_float lnzd;
30
31 PRINTF (("\nAMD version %d.%d.%d, %s, results:\n", AMD_MAIN_VERSION, AMD_SUB_VERSION, AMD_SUBSUB_VERSION, AMD_DATE)) ;
32
33 if (!ABIPInfo)
34 {
35 return ;
36 }
37
38 n = ABIPInfo [AMD_N] ;
39 ndiv = ABIPInfo [AMD_NDIV] ;
40 nmultsubs_ldl = ABIPInfo [AMD_NMULTSUBS_LDL] ;
41 nmultsubs_lu = ABIPInfo [AMD_NMULTSUBS_LU] ;
42 lnz = ABIPInfo [AMD_LNZ] ;
43 lnzd = (n >= 0 && lnz >= 0) ? (n + lnz) : (-1) ;
44
45 /* AMD return status */
46 PRINTF ((" status: ")) ;
47 if (ABIPInfo [AMD_STATUS] == AMD_OK)
48 {
49 PRINTF (("OK\n")) ;
50 }
51 else if (ABIPInfo [AMD_STATUS] == AMD_OUT_OF_MEMORY)
52 {
53 PRINTF (("out of memory\n")) ;
54 }
55 else if (ABIPInfo [AMD_STATUS] == AMD_INVALID)
56 {
57 PRINTF (("invalid matrix\n")) ;
58 }
59 else if (ABIPInfo [AMD_STATUS] == AMD_OK_BUT_JUMBLED)
60 {
61 PRINTF (("OK, but jumbled\n")) ;
62 }
63 else
64 {
65 PRINTF (("unknown\n")) ;
66 }
67
68 /* statistics about the input matrix */
69 PRI (" n, dimension of A: %.20g\n", n);
70 PRI (" nz, number of nonzeros in A: %.20g\n", ABIPInfo [AMD_NZ]) ;
71 PRI (" symmetry of A: %.4f\n", ABIPInfo [AMD_SYMMETRY]) ;
72 PRI (" number of nonzeros on diagonal: %.20g\n", ABIPInfo [AMD_NZDIAG]) ;
73 PRI (" nonzeros in pattern of A+A' (excl. diagonal): %.20g\n", ABIPInfo [AMD_NZ_A_PLUS_AT]) ;
74 PRI (" # dense rows/columns of A+A': %.20g\n", ABIPInfo [AMD_NDENSE]) ;
75
76 /* statistics about AMD's behavior */
77 PRI (" memory used, in bytes: %.20g\n", ABIPInfo [AMD_MEMORY]) ;
78 PRI (" # of memory compactions: %.20g\n", ABIPInfo [AMD_NCMPA]) ;
79
80 /* statistics about the ordering quality */
81 PRINTF (("\n"
82 " The following approximate statistics are for a subsequent\n"
83 " factorization of A(P,P) + A(P,P)'. They are slight upper\n"
84 " bounds if there are no dense rows/columns in A+A', and become\n"
85 " looser if dense rows/columns exist.\n\n")) ;
86
87 PRI (" nonzeros in L (excluding diagonal): %.20g\n", lnz) ;
88 PRI (" nonzeros in L (including diagonal): %.20g\n", lnzd) ;
89 PRI (" # divide operations for LDL' or LU: %.20g\n", ndiv) ;
90 PRI (" # multiply-subtract operations for LDL': %.20g\n", nmultsubs_ldl) ;
91 PRI (" # multiply-subtract operations for LU: %.20g\n", nmultsubs_lu) ;
92 PRI (" max nz. in any column of L (incl. diagonal): %.20g\n", ABIPInfo [AMD_DMAX]) ;
93
94 /* total flop counts for various factorizations */
95
96 if (n >= 0 && ndiv >= 0 && nmultsubs_ldl >= 0 && nmultsubs_lu >= 0)
97 {
98 PRINTF (("\n"
99 " chol flop count for real A, sqrt counted as 1 flop: %.20g\n"
100 " LDL' flop count for real A: %.20g\n"
101 " LDL' flop count for complex A: %.20g\n"
102 " LU flop count for real A (with no pivoting): %.20g\n"
103 " LU flop count for complex A (with no pivoting): %.20g\n\n",
104 n + ndiv + 2*nmultsubs_ldl,
105 ndiv + 2*nmultsubs_ldl,
106 9*ndiv + 8*nmultsubs_ldl,
107 ndiv + 2*nmultsubs_lu,
108 9*ndiv + 8*nmultsubs_lu)) ;
109 }
110}
