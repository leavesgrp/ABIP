35/* The SuiteSparse_config struct is available to all SuiteSparse functions and
36 to all applications that use those functions. It must be modified with
37 care, particularly in a multithreaded context. Normally, the application

39 by application-specific modifications if the applications wants to use
40 alternative memory manager functions.

47 SuiteSparse_start.

184 size_t nitems, /* number of items to malloc */

186)

235/* -------------------------------------------------------------------------- */
236/* SuiteSparse_realloc: realloc wrapper */
237/* -------------------------------------------------------------------------- */
238
239/* If p is non-NULL on input, it points to a previously allocated object of
240 size nitems_old * size_of_item. The object is reallocated to be of size
241 nitems_new * size_of_item. If p is NULL on input, then a new object of that
242 size is allocated. On success, a pointer to the new object is returned,
243 and ok is returned as 1. If the allocation fails, ok is set to 0 and a
244 pointer to the old (unmodified) object is returned.
245 */
246
247void *SuiteSparse_realloc /* pointer to reallocated block of memory, or to original block if the realloc failed. */
248(
249 size_t nitems_new, /* new number of items in the object */
250 size_t nitems_old, /* old number of items in the object */
251 size_t size_of_item, /* sizeof each item */
252 void *p, /* old object to reallocate */
253 int *ok /* 1 if successful, 0 otherwise */
254)
255{
256 size_t size ;
257 if (nitems_old < 1) nitems_old = 1 ;
258 if (nitems_new < 1) nitems_new = 1 ;
259 if (size_of_item < 1) size_of_item = 1 ;
260 size = nitems_new * size_of_item ;
261
262 if (size != ((abip_float) nitems_new) * size_of_item)
263 {
264 /* size_t overflow */
265 (*ok) = 0 ;
266 }
267 else if (p == ABIP_NULL)
268 {
269 /* a fresh object is being allocated */
270 p = SuiteSparse_malloc (nitems_new, size_of_item) ;
271 (*ok) = (p != ABIP_NULL) ;
272 }
273 else if (nitems_old == nitems_new)
274 {
275 /* the object does not change; do nothing */
276 (*ok) = 1 ;
277 }
278 else
279 {
280 /* change the size of the object from nitems_old to nitems_new */
281 void *pnew ;
282 pnew = (void *) (SuiteSparse_config.realloc_func) (p, size) ;
283
284 if (pnew == ABIP_NULL)
285 {
286 if (nitems_new < nitems_old)
287 {
288 /* the attempt to reduce the size of the block failed, but the old block is unchanged. So pretend to succeed. */
289 (*ok) = 1 ;
290 }
291 else
292 {
293 /* out of memory */
294 (*ok) = 0 ;
295 }
296 }
297 else
298 {
299 /* success */
300 p = pnew ;
301 (*ok) = 1 ;
302 }
303 }
304 return (p) ;
305}

325/* SuiteSparse_tic: return current wall clock time */
326/* -------------------------------------------------------------------------- */
327
328/* Returns the number of seconds (tic [0]) and nanoseconds (tic [1]) since some
329 * unspecified but fixed time in the past. If no timer is installed, zero is
330 * returned. A scalar abip_float precision value for 'tic' could be used, but this
331 * might cause loss of precision because clock_getttime returns the time from
332 * some distant time in the past. Thus, an array of size 2 is used.
333 *
334 * The timer is enabled by default. To disable the timer, compile with
335 * -DNTIMER. If enabled on a POSIX C 1993 system, the timer requires linking
336 * with the -lrt library.
337 *
338 * example:
339 *
340 * abip_float tic [2], r, s, t ;
341 * SuiteSparse_tic (tic) ; // start the timer
342 * // do some work A
343 * t = SuiteSparse_toc (tic) ; // t is time for work A, in seconds
344 * // do some work B
345 * s = SuiteSparse_toc (tic) ; // s is time for work A and B, in seconds
346 * SuiteSparse_tic (tic) ; // restart the timer
347 * // do some work C
348 * r = SuiteSparse_toc (tic) ; // s is time for work C, in seconds
349 *
350 * A abip_float array of size 2 is used so that this routine can be more easily
351 * ported to non-POSIX systems. The caller does not rely on the POSIX
352 * <time.h> include file.
353 */
354
355#ifdef SUITESPARSE_TIMER_ENABLED
356
357#include <time.h>
358
359void SuiteSparse_tic
360(
361 abip_float tic [2] /* output, contents undefined on input */
362)
363{
364 /* POSIX C 1993 timer, requires -librt */
365 struct timespec t ;
366 clock_gettime (CLOCK_MONOTONIC, &t) ;
367 tic [0] = (abip_float) (t.tv_sec) ;
368 tic [1] = (abip_float) (t.tv_nsec) ;
369}
370
371#else
372
373void SuiteSparse_tic
374(
375 abip_float tic [2] /* output, contents undefined on input */
376)
377{
378 /* no timer installed */
379 tic [0] = 0 ;
380 tic [1] = 0 ;
381}
382
383#endif

386/* -------------------------------------------------------------------------- */
387/* SuiteSparse_toc: return time since last tic */
388/* -------------------------------------------------------------------------- */
389
390/* Assuming SuiteSparse_tic is accurate to the nanosecond, this function is
391 * accurate down to the nanosecond for 2^53 nanoseconds since the last call to
392 * SuiteSparse_tic, which is sufficient for SuiteSparse (about 104 days). If
393 * additional accuracy is required, the caller can use two calls to
394 * SuiteSparse_tic and do the calculations differently.
395 */
396
397abip_float SuiteSparse_toc /* returns time in seconds since last tic */
398(
399 abip_float tic [2] /* input, not modified from last call to SuiteSparse_tic */
400)
401{
402 abip_float toc [2] ;
403 SuiteSparse_tic (toc) ;
404 return ((toc [0] - tic [0]) + 1e-9 * (toc [1] - tic [1])) ;
405}

443/* -------------------------------------------------------------------------- */
444/* SuiteSparse_hypot */
445/* -------------------------------------------------------------------------- */
446
447/* There is an equivalent routine called hypot in <math.h>, which conforms
448 * to ANSI C99. However, SuiteSparse does not assume that ANSI C99 is
449 * available. You can use the ANSI C99 hypot routine with:
450 *
451 * #include <math.h>
452 *i SuiteSparse_config.hypot_func = hypot ;
453 *
454 * Default value of the SuiteSparse_config.hypot_func pointer is
455 * SuiteSparse_hypot, defined below.
456 *
457 * s = hypot (x,y) computes s = sqrt (x*x + y*y) but does so more accurately.
458 * The NaN cases for the abip_float relops x >= y and x+y == x are safely ignored.
459 *
460 * Source: Algorithm 312, "Absolute value and square root of a complex number,"
461 * P. Friedland, Comm. ACM, vol 10, no 10, October 1967, page 665.
462 */
463
464abip_float SuiteSparse_hypot (abip_float x, abip_float y)
465{
466 abip_float s;
467 abip_float r ;
468 x = fabs (x) ;
469 y = fabs (y) ;
470
471 if (x >= y)
472 {
473 if (x + y == x)
474 {
475 s = x ;
476 }
477 else
478 {
479 r = y / x ;
480 s = x * sqrt (1.0 + r*r) ;
481 }
482 }
483 else
484 {
485 if (y + x == y)
486 {
487 s = y ;
488 }
489 else
490 {
491 r = x / y ;
492 s = y * sqrt (1.0 + r*r) ;
493 }
494 }
495 return (s) ;
496}

498/* -------------------------------------------------------------------------- */
499/* SuiteSparse_divcomplex */
500/* -------------------------------------------------------------------------- */
501
502/* c = a/b where c, a, and b are complex. The real and imaginary parts are
503 * passed as separate arguments to this routine. The NaN case is ignored
504 * for the abip_float relop br >= bi. Returns 1 if the denominator is zero,
505 * 0 otherwise.
506 *
507 * This uses ACM Algo 116, by R. L. Smith, 1962, which tries to avoid
508 * underflow and overflow.
509 *
510 * c can be the same variable as a or b.
511 *
512 * Default value of the SuiteSparse_config.divcomplex_func pointer is
513 * SuiteSparse_divcomplex.
514 */
515
516int SuiteSparse_divcomplex
517(
518 abip_float ar,
519 abip_float ai, /* real and imaginary parts of a */
520 abip_float br,
521 abip_float bi, /* real and imaginary parts of b */
522 abip_float *cr,
523 abip_float *ci /* real and imaginary parts of c */
524)
525{
526 abip_float tr;
527 abip_float ti;
528 abip_float r;
529 abip_float den;
530
531 if (fabs (br) >= fabs (bi))
532 {
533 r = bi / br ;
534 den = br + r * bi ;
535 tr = (ar + ai * r) / den ;
536 ti = (ai - ar * r) / den ;
537 }
538 else
539 {
540 r = br / bi ;
541 den = r * br + bi ;
542 tr = (ar * r + ai) / den ;
543 ti = (ai * r - ar) / den ;
544 }
545
546 *cr = tr ;
547 *ci = ti ;
548 return (den == 0.) ;
549}
