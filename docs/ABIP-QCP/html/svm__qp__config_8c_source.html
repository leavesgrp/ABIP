628void calc_svmqp_residuals(svmqp *self, ABIPWork *w, ABIPResiduals *r,
629 abip_int ipm_iter, abip_int admm_iter) {
630 DEBUG_FUNC
631
632 abip_int n = w->n;
633 abip_int m = w->m;
634
635 abip_float *y = (abip_float *)abip_malloc(m * sizeof(abip_float));
636 abip_float *x = (abip_float *)abip_malloc(n * sizeof(abip_float));
637 abip_float *s = (abip_float *)abip_malloc(n * sizeof(abip_float));
638
639 abip_float this_pr;
640 abip_float this_dr;
641 abip_float this_gap;
642
643 if (admm_iter && r->last_admm_iter == admm_iter) {
644 RETURN;
645 }
646
647 r->last_ipm_iter = ipm_iter;
648 r->last_admm_iter = admm_iter;
649
650 r->tau = ABS(w->u[n + m]);
651 r->kap =
652 ABS(w->v_origin[n + m]) /
653 (self->stgs->normalize ? (self->stgs->scale * self->sc_c * self->sc_b)
654 : 1);
655
656 memcpy(y, w->u, m * sizeof(abip_float));
657 memcpy(x, &w->u[m], n * sizeof(abip_float));
658 memcpy(s, &w->v_origin[m], n * sizeof(abip_float));
659
660 ABIP(scale_array)(y, 1 / r->tau, m);
661 ABIP(scale_array)(x, 1 / r->tau, n);
662 ABIP(scale_array)(s, 1 / r->tau, n);
663
664 abip_float *Ax = (abip_float *)abip_malloc(m * sizeof(abip_float));
665 abip_float *Ax_b = (abip_float *)abip_malloc(m * sizeof(abip_float));
666
667 memset(Ax, 0, m * sizeof(abip_float));
668 self->spe_A_times(self, x, Ax);
669
670 memcpy(Ax_b, Ax, m * sizeof(abip_float));
671 ABIP(add_scaled_array)(Ax_b, self->b, m, -1);
672
673 r->Ax_b_norm = ABIP(norm_inf)(Ax_b, m);
674
675 ABIP(c_dot)(Ax, self->D, m);
676 ABIP(c_dot)(Ax_b, self->D, m);
677
678 this_pr = ABIP(norm_inf)(Ax_b, m) /
679 (self->sc_b + MAX(ABIP(norm_inf)(Ax, m), self->sc_b * w->nm_inf_b));
680
681 abip_float *Qx = (abip_float *)abip_malloc(n * sizeof(abip_float));
682 abip_float *ATy = (abip_float *)abip_malloc(n * sizeof(abip_float));
683 abip_float *Qx_ATy_c_s = (abip_float *)abip_malloc(n * sizeof(abip_float));
684
685 memset(Qx, 0, n * sizeof(abip_float));
686 abip_float xQx_2 = 0;
687
688 if (self->Q != ABIP_NULL) {
689 ABIP(accum_by_A)(self->Q, x, Qx);
690 xQx_2 = ABIP(dot)(x, Qx, n) / (2 * self->sc_b * self->sc_c);
691 }
692
693 memset(ATy, 0, n * sizeof(abip_float));
694 self->spe_AT_times(self, y, ATy);
695
696 memcpy(Qx_ATy_c_s, Qx, n * sizeof(abip_float));
697 ABIP(add_scaled_array)(Qx_ATy_c_s, ATy, n, -1);
698 ABIP(add_scaled_array)(Qx_ATy_c_s, self->c, n, 1);
699 ABIP(add_scaled_array)(Qx_ATy_c_s, s, n, -1);
700
701 r->Qx_ATy_c_s_norm = ABIP(norm_inf)(Qx_ATy_c_s, n);
702
703 ABIP(c_dot)(Qx, self->E, n);
704 ABIP(c_dot)(ATy, self->E, n);
705 ABIP(c_dot)(Qx_ATy_c_s, self->E, n);
706 ABIP(c_dot)(s, self->E, n);
707
708 this_dr = ABIP(norm_inf)(Qx_ATy_c_s, n) /
709 (self->sc_c + MAX(self->sc_c * w->nm_inf_c, ABIP(norm_inf)(Qx, n)));
710
711 abip_float cTx = ABIP(dot)(self->c, x, n) / (self->sc_b * self->sc_c);
712 abip_float bTy = ABIP(dot)(self->b, y, m) / (self->sc_b * self->sc_c);
713
714 this_gap = ABS(2 * xQx_2 + cTx - bTy) /
715 (1 + MAX(2 * xQx_2, MAX(ABS(cTx), ABS(bTy))));
716
717 r->pobj = xQx_2 + cTx;
718 r->dobj = -xQx_2 + bTy;
719
720 r->res_dif = MAX(MAX(ABS(this_pr - r->res_pri), ABS(this_dr - r->res_dual)),
721 ABS(this_gap - r->rel_gap));
722 r->res_pri = this_pr;
723 r->res_dual = this_dr;
724 r->rel_gap = this_gap;
725 r->error_ratio =
726 MAX(r->res_pri / self->stgs->eps_p,
727 MAX(r->res_dual / self->stgs->eps_d, r->rel_gap / self->stgs->eps_g));
728
729 if (ABIP(dot)(self->c, &w->u[m], n) < 0) {
730 ABIP(scale_array)(Qx, r->tau, n);
731 ABIP(scale_array)(Ax, r->tau, m);
732 r->res_unbdd = MAX(ABIP(norm)(Qx, n), ABIP(norm)(Ax, m)) /
733 (-ABIP(dot)(self->c, &w->u[m], n));
734 } else {
735 r->res_unbdd = INFINITY;
736 }
737
738 if (ABIP(dot)(self->b, w->u, m) > 0) {
739 ABIP(scale_array)(ATy, r->tau, n);
740 ABIP(scale_array)(s, r->tau, n);
741 ABIP(add_scaled_array)(ATy, s, n, 1);
742
743 r->res_infeas = ABIP(norm)(ATy, n) / ABIP(dot)(self->b, w->u, m);
744 } else {
745 r->res_infeas = INFINITY;
746 }
747
748 abip_free(x);
749 abip_free(y);
750 abip_free(s);
751 abip_free(Ax);
752 abip_free(Ax_b);
753 abip_free(Qx);
754 abip_free(ATy);
755 abip_free(Qx_ATy_c_s);
756}

761cs *form_svmqp_kkt(svmqp *self) {
762 abip_int n = self->n + 1;
763 abip_int m = self->m;
764 cs *LTL;
765
766 cs *B1 = cs_spalloc(m, n, self->A->p[n], 1, 0);
767 memcpy(B1->i, self->A->i, self->A->p[n] * sizeof(abip_int));
768 memcpy(B1->p, self->A->p, (n + 1) * sizeof(abip_int));
769 memcpy(B1->x, self->A->x, self->A->p[n] * sizeof(abip_float));
770
771 cs *B2 = cs_spalloc(m, n, self->A->p[n], 1, 0);
772 memcpy(B2->i, B1->i, B1->p[n] * sizeof(abip_int));
773 memcpy(B2->p, B1->p, (n + 1) * sizeof(abip_int));
774 memcpy(B2->x, B1->x, B1->p[n] * sizeof(abip_float));
775
776 if (m > n) {
777 cs *T1 = cs_spalloc(n, n, n, 1, 1);
778
779 for (int i = 0; i < n; i++) {
780 cs_entry(T1, i, i, self->H[i]);
781 }
782
783 cs *eye = cs_compress(T1);
784
785 cs_spfree(T1);
786
787 for (int i = 0; i < B1->p[n]; i++) {
788 B1->x[i] /= self->F[B1->i[i]];
789 }
790
791 LTL = cs_add(eye, cs_multiply(cs_transpose(B2, 1), B1), 1, 1);
792
793 cs_spfree(eye);
794 } else {
795 for (int i = 0; i < B2->n; i++) {
796 for (int j = B2->p[i]; j < B2->p[i + 1]; j++) {
797 B2->x[j] /= self->H[i];
798 }
799 }
800 cs *diag = cs_spalloc(m, m, m, 1, 1);
801
802 for (int i = 0; i < m; i++) {
803 cs_entry(diag, i, i, self->F[i]);
804 }
805 cs *diag_F = cs_compress(diag);
806 cs_spfree(diag);
807
808 LTL = cs_add(diag_F, cs_multiply(B2, cs_transpose(B1, 1)), 1, 1);
809 }
810 cs_spfree(B2);
811 cs_spfree(B1);
812
813 for (int i = 0; i < LTL->n; i++) {
814 for (int j = LTL->p[i]; j < LTL->p[i + 1]; j++) {
815 if (LTL->i[j] > i) LTL->x[j] = 0;
816 }
817 }
818 cs_dropzeros(LTL);
819 return LTL;
820}

868abip_int init_svmqp_linsys_work(svmqp *self) {
869 if (self->stgs->linsys_solver == 0) { // mkl need lower triangle
870 self->L->K = cs_transpose(form_svmqp_kkt(self), 1);
871 } else if (self->stgs->linsys_solver == 1) { // qdldl need upper triangle
872 self->L->K = form_svmqp_kkt(self);
873 } else if (self->stgs->linsys_solver == 2) { // cholesky need upper triangle
874 self->L->K = form_svmqp_kkt(self);
875 } else if (self->stgs->linsys_solver == 3) { // pcg doesn't need kkt matrix
876 init_svmqp_precon(self);
877 self->L->K = ABIP_NULL;
878 } else if (self->stgs->linsys_solver ==
879 4) { // mkl-pardiso need lower triangle
880 self->L->K = cs_transpose(form_svmqp_kkt(self), 1);
881 } else if (self->stgs->linsys_solver ==
882 5) { // dense cholesky need upper triangle
883 self->L->K = form_svmqp_kkt(self);
884 } else {
885 printf("\nlinsys solver type error\n");
886 return -1;
887 }
888 return ABIP(init_linsys_work)(self);

894abip_int solve_svmqp_linsys(svmqp *self, abip_float *b,
895 abip_float *pcg_warm_start, abip_int iter,
896 abip_float error_ratio) {
897 ABIP(timer) linsys_timer;
898 ABIP(tic)(&linsys_timer);
899 abip_int n = self->n;
900 abip_int m = self->m;
901
902 abip_int p = self->p;
903 abip_int q = self->q;
904
905 if (self->stgs->linsys_solver == 3) { // pcg
906
907 abip_int n = self->q;
908 abip_int m = self->p;
909 abip_int i;
910
911 abip_float norm_p = ABIP(norm)(&b[m], n);
912
913 abip_float *tem = (abip_float *)abip_malloc(sizeof(abip_float) * m);
914 memcpy(tem, b, m * sizeof(abip_float));
915 for (i = 0; i < m; i++) {
916 tem[i] /= self->rho_dr[i];
917 }
918 self->spe_AT_times(self, tem, &b[m]);
919
920 abip_free(tem);
921
922 abip_float pcg_tol = get_svmqp_pcg_tol(iter, error_ratio, norm_p);
923 abip_int cg_its =
924 ABIP(solve_linsys)(self, &b[m], n, &pcg_warm_start[m], pcg_tol);
925
926 if (iter >= 0) {
927 self->L->total_cg_iters += cg_its;
928 }
929
930 ABIP(scale_array)(b, -1, m);
931 self->spe_A_times(self, &b[m], b);
932 for (i = 0; i < m; i++) {
933 b[i] /= -self->rho_dr[i];
934 }
935 } else { // direct methods
936
937 abip_float *b2 = (abip_float *)abip_malloc(p * sizeof(abip_float));
938 memcpy(b2, b, p * sizeof(abip_float));
939 abip_float *tmp = (abip_float *)abip_malloc(q * sizeof(abip_float));
940 memcpy(tmp, &b[p], sizeof(abip_float) * q);
941
942 for (int i = 0; i < q; i++) {
943 tmp[i] /= -self->H[i];
944 }
945
946 self->spe_A_times(self, tmp, b2);
947
948 if (m > n + 1 && self->stgs->linsys_solver != 3) {
949 for (int i = 0; i < p; i++) {
950 b2[i] /= self->F[i];
951 }
952
953 abip_float *tmp1 =
954 (abip_float *)abip_malloc((n + 1) * sizeof(abip_float));
955 memset(tmp1, 0, (n + 1) * sizeof(abip_float));
956
957 ABIP(accum_by_Atrans)(self->A, b2, tmp1);
958
959 ABIP(solve_linsys)(self, tmp1, n + 1, ABIP_NULL, 0);
960
961 abip_float *tmp2 = (abip_float *)abip_malloc(m * sizeof(abip_float));
962 memset(tmp2, 0, m * sizeof(abip_float));
963
964 ABIP(accum_by_A)(self->A, tmp1, tmp2);
965
966 for (int i = 0; i < m; i++) {
967 tmp2[i] /= self->F[i];
968 }
969
970 ABIP(add_scaled_array)(b2, tmp2, m, -1);
971
972 abip_free(tmp1);
973 abip_free(tmp2);
974 } else {
975 abip_int cg_its =
976 ABIP(solve_linsys)(self, b2, m, pcg_warm_start, error_ratio);
977 if (iter >= 0) {
978 self->L->total_cg_iters += cg_its;
979 }
980 }
981
982 memcpy(b, b2, m * sizeof(abip_float));
983 abip_free(b2);
984 abip_free(tmp);
985
986 } // if pcg
987
988 self->spe_AT_times(self, b, &b[p]);
989
990 for (int i = 0; i < q; i++) {
991 b[p + i] /= self->H[i];
992 }
993
994 self->L->total_solve_time += ABIP(tocq)(&linsys_timer);
995
996 return 0;
997}
