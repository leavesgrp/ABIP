518abip_float qcp_inner_conv_check(qcp *self, ABIPWork *w) {
519 abip_int m = self->p;
520 abip_int n = self->q;
521
522 abip_float *Qu = (abip_float *)abip_malloc((m + n + 1) * sizeof(abip_float));
523 abip_float *Mu = (abip_float *)abip_malloc((m + n) * sizeof(abip_float));
524
525 memset(Mu, 0, (m + n) * sizeof(abip_float));
526
527 self->spe_A_times(self, &w->u[m], Mu);
528
529 self->spe_AT_times(self, w->u, &Mu[m]);
530 ABIP(scale_array)(&Mu[m], -1, n);
531
532 if (self->Q != ABIP_NULL) {
533 ABIP(accum_by_A)(self->Q, &w->u[m], &Mu[m]);
534 }
535
536 memcpy(Qu, Mu, (m + n) * sizeof(abip_float));
537
538 ABIP(add_scaled_array)(Qu, self->b, m, -w->u[m + n]);
539 ABIP(add_scaled_array)(&Qu[m], self->c, n, w->u[m + n]);
540
541 Qu[m + n] = -ABIP(dot)(w->u, Mu, m + n) / w->u[m + n] +
542 ABIP(dot)(w->u, self->b, m) - ABIP(dot)(&w->u[m], self->c, n);
543
544 abip_float *tem = (abip_float *)abip_malloc((m + n + 1) * sizeof(abip_float));
545 memcpy(tem, Qu, (m + n + 1) * sizeof(abip_float));
546 ABIP(add_scaled_array)(tem, w->v_origin, m + n + 1, -1);
547
548 abip_float error_inner =
549 ABIP(norm)(tem, m + n + 1) /
550 (1 + ABIP(norm)(Qu, m + n + 1) + ABIP(norm)(w->v_origin, m + n + 1));
551
552 abip_free(Qu);
553 abip_free(Mu);
554 abip_free(tem);
555
556 return error_inner;
557}

631 memcpy(Qx_ATy_c_s, Qx, n * sizeof(abip_float));
632 ABIP(add_scaled_array)(Qx_ATy_c_s, ATy, n, -1);
633 ABIP(add_scaled_array)(Qx_ATy_c_s, self->c, n, 1);
634 ABIP(add_scaled_array)(Qx_ATy_c_s, s, n, -1);
635
636 r->Qx_ATy_c_s_norm = ABIP(norm_inf)(Qx_ATy_c_s, n);
637
638 ABIP(c_dot)(Qx, self->E, n);
639 ABIP(c_dot)(ATy, self->E, n);
640 ABIP(c_dot)(Qx_ATy_c_s, self->E, n);
641 ABIP(c_dot)(s, self->E, n);
642
643 this_dr = ABIP(norm_inf)(Qx_ATy_c_s, n) /
644 (self->sc_c + MAX(self->sc_c * w->nm_inf_c, ABIP(norm_inf)(Qx, n)));
645
646 abip_float cTx = ABIP(dot)(self->c, x, n) / (self->sc_b * self->sc_c);
647 abip_float bTy = ABIP(dot)(self->b, y, m) / (self->sc_b * self->sc_c);
648
649 this_gap = ABS(2 * xQx_2 + cTx - bTy) /
650 (1 + MAX(2 * xQx_2, MAX(ABS(cTx), ABS(bTy))));
651
652 r->pobj = xQx_2 + cTx;
653 r->dobj = -xQx_2 + bTy;
654
655 r->res_dif = MAX(MAX(ABS(this_pr - r->res_pri), ABS(this_dr - r->res_dual)),
656 ABS(this_gap - r->rel_gap));
657 r->res_pri = this_pr;
658 r->res_dual = this_dr;
659 r->rel_gap = this_gap;
660 r->error_ratio =
661 MAX(r->res_pri / self->stgs->eps_p,
662 MAX(r->res_dual / self->stgs->eps_d, r->rel_gap / self->stgs->eps_g));
663
664 if (ABIP(dot)(self->c, &w->u[m], n) < 0) {
665 ABIP(scale_array)(Qx, r->tau, n);
666 ABIP(scale_array)(Ax, r->tau, m);
667 r->res_unbdd = MAX(ABIP(norm)(Qx, n), ABIP(norm)(Ax, m)) /
668 (-ABIP(dot)(self->c, &w->u[m], n));
669 } else {
670 r->res_unbdd = INFINITY;
671 }
672
673 if (ABIP(dot)(self->b, w->u, m) > 0) {
674 ABIP(scale_array)(ATy, r->tau, n);
675 ABIP(scale_array)(s, r->tau, n);
676 ABIP(add_scaled_array)(ATy, s, n, 1);
677
678 r->res_infeas = ABIP(norm)(ATy, n) / ABIP(dot)(self->b, w->u, m);
679 } else {
680 r->res_infeas = INFINITY;
681 }
682
683 abip_free(x);
684 abip_free(y);
685 abip_free(s);
686 abip_free(Ax);
687 abip_free(Ax_b);
688 abip_free(Qx);
689 abip_free(ATy);
690 abip_free(Qx_ATy_c_s);
691}

696/*K = -rho_dr(1:m)*I -A
697 Q + rho_dr(m+1:m+n)*I
698*/
699cs *form_qcp_kkt(qcp *self) {
700 abip_int m = self->m;
701 abip_int n = self->n;
702 ABIPMatrix *A = self->A;
703 ABIPMatrix *Q = self->Q;
704
705 abip_int nnzA = A == ABIP_NULL ? 0 : A->p[A->n];
706 ;
707 abip_int nnzQ = Q == ABIP_NULL ? 0 : Q->p[Q->n];
708 abip_int nnzK = m + nnzA + nnzQ + n;
709 abip_int i;
710 abip_int j;
711 abip_float tem;
712
713 cs *K = cs_spalloc(m + n, m + n, nnzK, 1, 1);
714
715 for (i = 0; i < m; i++) {
716 cs_entry(K, i, i, -self->rho_dr[i]);
717 }
718
719 if (A != ABIP_NULL) {
720 for (i = 0; i < n; i++) {
721 for (j = A->p[i]; j < A->p[i + 1]; j++) {
722 cs_entry(K, A->i[j], m + i, -A->x[j]);
723 }
724 }
725 }
726
727 for (i = 0; i < n; i++) {
728 if (Q == ABIP_NULL || Q->p[i] == Q->p[i + 1]) {
729 cs_entry(K, m + i, m + i, self->rho_dr[m + i]);
730 } else {
731 for (j = Q->p[i]; j < Q->p[i + 1]; j++) {
732 if (Q->i[j] > i)
733 tem = 0;
734 else if (Q->i[j] == i)
735 tem = Q->x[j] + self->rho_dr[m + i];
736 else
737 tem = Q->x[j];
738 cs_entry(K, m + Q->i[j], m + i, tem);
739 }
740 }
741 }
742
743 cs *K_csc = cs_compress(K);
744 cs_spfree(K);
745 cs_dropzeros(K_csc);
746
747 return K_csc;
748}

754void init_qcp_precon(qcp *self) {
755 self->L->M = (abip_float *)abip_malloc(self->q * sizeof(abip_float));
756 memset(self->L->M, 0, self->q * sizeof(abip_float));
757
758 for (int i = 0; i < self->q; i++) {
759 for (int j = self->A->p[i]; j < self->A->p[i + 1]; j++) {
760 self->L->M[i] +=
761 self->A->x[j] * self->A->x[j] / self->rho_dr[self->A->i[j]];
762 }
763 }
764
765 if (self->Q != ABIP_NULL) {
766 for (int i = 0; i < self->q; i++) {
767 for (int j = self->Q->p[i]; j < self->Q->p[i + 1]; j++) {
768 if (i == self->Q->i[j]) {
769 self->L->M[i] += self->Q->x[j];
770 break;
771 }
772 }
773 }
774 }
775
776 ABIP(add_scaled_array)(self->L->M, &self->rho_dr[self->p], self->q, 1);
777 for (int i = 0; i < self->q; i++) {
778 self->L->M[i] = 1.0 / self->L->M[i];
779 }
780}

800 if (self->stgs->linsys_solver == 0) { // mkl-dss need lower triangle
801 self->L->K = cs_transpose(form_qcp_kkt(self), 1);
802 } else if (self->stgs->linsys_solver == 1) { // qdldl need upper triangle

805 self->L->K = form_qcp_kkt(self);

812 } else if (self->stgs->linsys_solver ==
813 5) { // dense cholesky need upper triangle
814 self->L->K = form_qcp_kkt(self);
815 } else {
816 printf("\nlinsys solver type error\n");
817 return -1;
818 }
819
820 return ABIP(init_linsys_work)(self);

826abip_int solve_qcp_linsys(qcp *self, abip_float *b, abip_float *pcg_warm_start,
827 abip_int iter, abip_float error_ratio) {

830
831 if (self->stgs->linsys_solver == 3) { // pcg
832
833 abip_int n = self->n;
834 abip_int m = self->m;
835 abip_int i;
836

838

842 tem[i] /= self->rho_dr[i];

846 abip_free(tem);

849
850 abip_int cg_its;
851 if (iter == -1) {
852 cg_its = ABIP(solve_linsys)(self, &b[m], n, pcg_warm_start, error_ratio);
853 } else {
854 cg_its =
855 ABIP(solve_linsys)(self, &b[m], n, &pcg_warm_start[m], error_ratio);
856 }
857
858 if (iter >= 0) {
859 self->L->total_cg_iters += cg_its;
860 }
861
862 ABIP(scale_array)(b, -1, m);
863 self->spe_A_times(self, &b[m], b);
864 for (i = 0; i < m; i++) {
865 b[i] /= -self->rho_dr[i];
866 }
867
868 } else { // direct methods
869
870 abip_int n = self->n + self->m;
871
872 // difference here
873 ABIP(scale_array)(b, -1, self->m);
874
875 ABIP(solve_linsys)(self, b, n, ABIP_NULL, 0);
876 }
877
878 self->L->total_solve_time += ABIP(tocq)(&linsys_timer);
879
880 return 0;
881}
