8abip_int init_lasso(lasso **self, ABIPData *d, ABIPSettings *stgs) {
9 lasso *this_lasso = (lasso *)abip_malloc(sizeof(lasso));
10 *self = this_lasso;
11
12 this_lasso->m = d->m;
13 this_lasso->n = d->n;
14 this_lasso->p = d->m + 1;
15 this_lasso->q = 2 + 2 * d->n + d->m;
16 abip_int m = this_lasso->p;
17 abip_int n = this_lasso->q;
18
19 this_lasso->lambda = d->lambda;
20 this_lasso->Q = ABIP_NULL;
21 this_lasso->sparsity = (((abip_float)d->A->p[d->n] / (d->m * d->n)) < 0.1);
22
23 // non-identity DR scaling
24 this_lasso->rho_dr =
25 (abip_float *)abip_malloc((m + n + 1) * sizeof(abip_float));
26 for (int i = 0; i < m + n + 1; i++) {
27 if (i < m) {
28 this_lasso->rho_dr[i] = stgs->rho_y;
29 } else if (i < m + n) {
30 this_lasso->rho_dr[i] = stgs->rho_x;
31 } else {
32 this_lasso->rho_dr[i] = stgs->rho_tau;
33 }
34 }
35
36 if (this_lasso->sparsity) {
37 this_lasso->sc = 2;
38 this_lasso->sc_c = 1 / d->lambda;
39 this_lasso->sc_cone2 = d->lambda / d->m * 80;
40 this_lasso->sc_cone1 = 0.8 / this_lasso->sc_c / this_lasso->sc_cone2;
41 this_lasso->sc_b = this_lasso->sc_c * 300 * d->lambda / d->m;
42 } else {
43 if (d->m < d->n)
44 this_lasso->sc = 4;
45 else
46 this_lasso->sc = 1;
47 this_lasso->sc_c = 1 / d->lambda;
48 this_lasso->sc_b = this_lasso->sc_c;
49 this_lasso->sc_cone2 = 0.8;
50 this_lasso->sc_cone1 = 1 / this_lasso->sc_c;
51 }
52
53 this_lasso->L = (ABIPLinSysWork *)abip_malloc(sizeof(ABIPLinSysWork));
54 this_lasso->pro_type = LASSO;
55 this_lasso->stgs = stgs;
56 this_lasso->data = d;
57
58 abip_float *data_b =
59 (abip_float *)abip_malloc(this_lasso->p * sizeof(abip_float));
60 data_b[0] = 1;
61 memcpy(&data_b[1], d->b, d->m * sizeof(abip_float));
62 this_lasso->data->b = data_b;
63
64 abip_float *data_c =
65 (abip_float *)abip_malloc(this_lasso->q * sizeof(abip_float));
66 data_c[0] = 0;
67 data_c[1] = 2;
68 memset(&data_c[2], 0, d->m * sizeof(abip_float));
69 for (int i = 0; i < 2 * d->n; i++) {
70 data_c[i + 2 + d->m] = d->lambda;
71 }
72 this_lasso->data->c = data_c;
73 /* for lasso problem, no need for inputing c*/
74 d->c = data_c;
75
76 this_lasso->A = (ABIPMatrix *)abip_malloc(sizeof(ABIPMatrix));
77 this_lasso->b = (abip_float *)abip_malloc(this_lasso->p * sizeof(abip_float));
78 this_lasso->c = (abip_float *)abip_malloc(this_lasso->q * sizeof(abip_float));
79 this_lasso->D = (abip_float *)abip_malloc(this_lasso->m * sizeof(abip_float));
80 this_lasso->E = (abip_float *)abip_malloc(this_lasso->n * sizeof(abip_float));
81
82 this_lasso->scaling_data = &scaling_lasso_data;
83 this_lasso->un_scaling_sol = &un_scaling_lasso_sol;
84 this_lasso->calc_residuals = &calc_lasso_residuals;
85 this_lasso->init_spe_linsys_work = &init_lasso_linsys_work;
86 this_lasso->solve_spe_linsys = &solve_lasso_linsys;

91
92 return 0;

99void lasso_A_times(lasso *self, const abip_float *x, abip_float *y) {
100 y[0] += x[0];
101
102 for (int i = 1; i < self->p; i++) {
103 y[i] += self->D[i - 1] * SQRTF(self->sc_cone2) * x[i + 1];
104 }
105
106 ABIP(accum_by_A)(self->A, &x[self->m + 2], &y[1]);
107 ABIP(scale_array)(&y[1], -1, self->m);
108 ABIP(accum_by_A)(self->A, &x[self->m + self->n + 2], &y[1]);
109 ABIP(scale_array)(&y[1], -1, self->m);
110}

116void lasso_AT_times(lasso *self, const abip_float *x, abip_float *y) {
117 y[0] += x[0];
118 for (int i = 2; i < self->m + 2; i++) {
119 y[i] += x[i - 1] * self->D[i - 2] * SQRTF(self->sc_cone2);
120 }
121
122 ABIP(accum_by_Atrans)(self->A, &x[1], &y[self->m + 2]);
123 ABIP(scale_array)(&y[self->m + 2 + self->n], -1, self->n);
124 ABIP(accum_by_Atrans)(self->A, &x[1], &y[self->m + 2 + self->n]);
125 ABIP(scale_array)(&y[self->m + 2 + self->n], -1, self->n);
126}

131void scaling_lasso_data(lasso *self, ABIPCone *k) {
132 if (!ABIP(copy_A_matrix)(&(self->A), self->data->A)) {
133 abip_printf("ERROR: copy A matrix failed\n");
134 RETURN;
135 }
136
137 abip_int m = self->m;
138 abip_int n = self->n;
139 ABIPMatrix *A = self->A;
140
141 abip_float min_row_scale = MIN_SCALE * SQRTF((abip_float)n);
142 abip_float max_row_scale = MAX_SCALE * SQRTF((abip_float)n);
143 abip_float min_col_scale = MIN_SCALE * SQRTF((abip_float)m);
144 abip_float max_col_scale = MAX_SCALE * SQRTF((abip_float)m);
145
146 abip_float *E = self->E;
147 memset(E, 0, n * sizeof(abip_float));
148
149 abip_float *D = self->D;
150 memset(D, 0, m * sizeof(abip_float));
151
152 abip_float avg = 0;
153 abip_float avg1 = 0;
154
155 if (self->stgs->scale_E) {
156 if (self->sparsity) {
157 for (int i = 0; i < n; i++) {
158 for (int j = A->p[i]; j < A->p[i + 1]; j++) {
159 E[i] += A->x[j] * A->x[j];
160 }
161
162 avg += SQRTF(E[i]);
163 }
164
165 avg /= n;
166
167 for (int i = 0; i < n; i++) {
168 E[i] = avg / SQRTF(E[i] + 1e-4) / self->sc;
169 if (E[i] > 1000 * SQRTF(m)) {
170 E[i] = 1000 * SQRTF(m);
171 }
172 if (E[i] < 0.001 * SQRTF(m)) {
173 E[i] = 1;
174 }
175 if (E[i] > 50) {
176 E[i] = 50;
177 }
178
179 avg1 += E[i];
180 }
181
182 avg1 /= n;
183
184 for (int i = 0; i < n; i++) {
185 E[i] = avg1 / E[i] / self->sc;
186 }
187 } else {
188 for (int i = 0; i < n; i++) {
189 for (int j = A->p[i]; j < A->p[i + 1]; j++) {
190 E[i] += A->x[j] * A->x[j];
191 }
192
193 E[i] = SQRTF(E[i]);
194
195 if (E[i] > 1000 * SQRTF(m)) {
196 E[i] = 1000 * SQRTF(m);
197 }
198 if (E[i] < 0.001 * SQRTF(m)) {
199 E[i] = 1;
200 }
201 if (E[i] > 7) {
202 E[i] = 7;
203 }
204
205 E[i] = 1 / (E[i] * self->sc);
206 }
207 }
208
209 for (int i = 0; i < n; i++) {
210 for (int j = A->p[i]; j < A->p[i + 1]; j++) {
211 A->x[j] *= E[i];
212 }
213 }
214 }
215
216 for (int i = 0; i < A->p[n]; i++) {
217 D[A->i[i]] += A->x[i] * A->x[i];
218 }
219
220 avg = 0;
221 for (int i = 0; i < m; i++) {
222 avg += SQRTF(2 * D[i] + self->sc_cone2);
223 }
224
225 avg /= m;
226
227 for (int i = 0; i < m; i++) {
228 D[i] = avg / SQRTF(2 * D[i] + self->sc_cone2);
229 }
230
231 for (int i = 0; i < A->p[n]; i++) {
232 A->x[i] *= D[A->i[i]];
233 }
234
235 memcpy(self->b, self->data->b, self->p * sizeof(abip_float));
236
237 self->b[0] = self->sc_cone1;
238
239 for (int i = 1; i < m + 1; i++) {
240 self->b[i] *= D[i - 1];
241 }
242 ABIP(scale_array)(self->b, self->sc_b, self->p);
243
244 self->c[0] = 0;
245 self->c[1] = self->sc_cone1 * self->sc_cone2;
246 memset(&self->c[2], 0, self->m * sizeof(abip_float));
247
248 for (int i = 0; i < self->n; i++) {
249 self->c[i + self->m + 2] = E[i] * self->lambda;
250 self->c[i + self->m + 2 + self->n] = E[i] * self->lambda;
251 }
252 ABIP(scale_array)(self->c, self->sc_c, self->q);
253
254 self->D_hat = (abip_float *)abip_malloc(m * sizeof(abip_float));
255
256 for (int i = 0; i < m; i++) {
257 self->D_hat[i] =
258 self->D[i] * self->D[i] * self->sc_cone2 + self->stgs->rho_y;
259 }
260}

323abip_float lasso_inner_conv_check(lasso *self, ABIPWork *w) {
324 abip_int m = self->p;
325 abip_int n = self->q;
326
327 abip_float *Qu = (abip_float *)abip_malloc((m + n + 1) * sizeof(abip_float));
328 abip_float *Mu = (abip_float *)abip_malloc((m + n) * sizeof(abip_float));
329
330 memset(Mu, 0, (m + n) * sizeof(abip_float));
331
332 self->spe_A_times(self, &w->u[m], Mu);
333
334 self->spe_AT_times(self, w->u, &Mu[m]);
335 ABIP(scale_array)(&Mu[m], -1, n);
336
337 if (self->Q != ABIP_NULL) {
338 ABIP(accum_by_A)(self->Q, &w->u[m], &Mu[m]);
339 }
340
341 memcpy(Qu, Mu, (m + n) * sizeof(abip_float));
342
343 ABIP(add_scaled_array)(Qu, self->b, m, -w->u[m + n]);
344 ABIP(add_scaled_array)(&Qu[m], self->c, n, w->u[m + n]);
345
346 Qu[m + n] = -ABIP(dot)(w->u, Mu, m + n) / w->u[m + n] +
347 ABIP(dot)(w->u, self->b, m) - ABIP(dot)(&w->u[m], self->c, n);
348
349 abip_float *tem = (abip_float *)abip_malloc((m + n + 1) * sizeof(abip_float));
350 memcpy(tem, Qu, (m + n + 1) * sizeof(abip_float));
351 ABIP(add_scaled_array)(tem, w->v_origin, m + n + 1, -1);
352
353 abip_float error_inner =
354 ABIP(norm)(tem, m + n + 1) /
355 (1 + ABIP(norm)(w->u, m + n + 1) + ABIP(norm)(w->v_origin, m + n + 1));
356
357 abip_free(Qu);
358 abip_free(Mu);
359 abip_free(tem);
360
361 return error_inner;
362}

367void calc_lasso_residuals(lasso *self, ABIPWork *w, ABIPResiduals *r,
368 abip_int ipm_iter, abip_int admm_iter) {
369 abip_int p = w->m;
370 abip_int q = w->n;
371 abip_int m = self->m;
372 abip_int n = self->n;
373 abip_float this_pr;
374 abip_float this_dr;
375 abip_float this_gap;
376
377 r->tau = w->u[p + q];
378
379 abip_float *x = (abip_float *)abip_malloc(m * sizeof(abip_float));
380 memcpy(x, &(w->u[m + 3]), m * sizeof(abip_float));
381 ABIP(scale_array)(x, SQRTF(self->sc_cone2) / (r->tau * self->sc_b), m);
382
383 abip_float *beta_plus = (abip_float *)abip_malloc(n * sizeof(abip_float));
384 memcpy(beta_plus, &(w->u[2 * m + 3]), n * sizeof(abip_float));
385 for (int i = 0; i < n; i++) {
386 beta_plus[i] *= self->E[i] / (r->tau * self->sc_b);
387 }
388
389 abip_float *beta_minus = (abip_float *)abip_malloc(n * sizeof(abip_float));
390 memcpy(beta_minus, &(w->u[2 * m + 3 + n]), n * sizeof(abip_float));
391 for (int i = 0; i < n; i++) {
392 beta_minus[i] *= self->E[i] / (r->tau * self->sc_b);
393 }
394
395 abip_float *z = (abip_float *)abip_malloc(m * sizeof(abip_float));
396 memcpy(z, &(w->u[1]), m * sizeof(abip_float));
397 for (int i = 0; i < m; i++) {
398 z[i] *= self->D[i] / (r->tau * self->sc_c);
399 }
400
401 abip_float *s1 = (abip_float *)abip_malloc(n * sizeof(abip_float));
402 memcpy(s1, &(w->v[2 * m + 3]), n * sizeof(abip_float));
403 for (int i = 0; i < n; i++) {
404 s1[i] /= self->E[i] * r->tau * self->sc_c;
405 }
406
407 abip_float *s2 = (abip_float *)abip_malloc(n * sizeof(abip_float));
408 memcpy(s2, &(w->v[2 * m + n + 3]), n * sizeof(abip_float));
409 for (int i = 0; i < n; i++) {
410 s2[i] /= self->E[i] * r->tau * self->sc_c;
411 }
412
413 abip_float *pr = (abip_float *)abip_malloc(m * sizeof(abip_float));
414 memcpy(pr, x, m * sizeof(abip_float));
415 ABIP(accum_by_A)(self->data->A, beta_plus, pr);
416 ABIP(scale_array)(pr, -1, m);
417 ABIP(accum_by_A)(self->data->A, beta_minus, pr);
418 ABIP(scale_array)(pr, -1, m);
419 ABIP(add_scaled_array)(pr, &self->data->b[1], m, -1);
420 this_pr = ABIP(norm)(pr, m) / MAX(ABIP(norm)(&self->data->b[1], m), 1);
421
422 abip_float *dr1 = (abip_float *)abip_malloc(n * sizeof(abip_float));
423 memcpy(dr1, s1, n * sizeof(abip_float));
424 ABIP(accum_by_Atrans)(self->data->A, z, dr1);
425 for (int i = 0; i < n; i++) {
426 dr1[i] -= self->lambda;
427 }
428
429 abip_float *dr2 = (abip_float *)abip_malloc(n * sizeof(abip_float));
430 memcpy(dr2, s2, n * sizeof(abip_float));
431 ABIP(scale_array)(dr2, -1, n);
432 ABIP(accum_by_Atrans)(self->data->A, z, dr2);
433 ABIP(scale_array)(dr2, -1, n);
434 for (int i = 0; i < n; i++) {
435 dr2[i] -= self->lambda;
436 }
437 abip_float *dr = (abip_float *)abip_malloc(2 * n * sizeof(abip_float));
438 memcpy(dr, dr1, n * sizeof(abip_float));
439 memcpy(&dr[n], dr2, n * sizeof(abip_float));
440 this_dr = ABIP(norm)(dr, 2 * n) / (SQRTF(2 * n) * self->lambda);
441
442 abip_float *lambda_ones = (abip_float *)abip_malloc(n * sizeof(abip_float));
443 for (int i = 0; i < n; i++) {
444 lambda_ones[i] = self->lambda;
445 }
446 this_gap =
447 ABS(0.5 * ABIP(dot)(x, x, m) + ABIP(dot)(lambda_ones, beta_plus, n) +
448 ABIP(dot)(lambda_ones, beta_minus, n) + 0.5 * ABIP(dot)(z, z, m) -
449 ABIP(dot)(&self->data->b[1], z, m)) /
450 (1 + ABS(0.5 * ABIP(dot)(x, x, m) + ABIP(dot)(lambda_ones, beta_plus, n) +
451 ABIP(dot)(lambda_ones, beta_minus, n)));
452
453 r->last_ipm_iter = ipm_iter;
454 r->last_admm_iter = admm_iter;
455
456 r->dobj = (-0.5 * ABIP(dot)(z, z, m) + ABIP(dot)(&self->data->b[1], z, m));
457 r->pobj = (0.5 * ABIP(dot)(x, x, m) + ABIP(dot)(lambda_ones, beta_plus, n) +
458 ABIP(dot)(lambda_ones, beta_minus, n));
459
460 r->res_dif = MAX(MAX(ABS(this_pr - r->res_pri), ABS(this_dr - r->res_dual)),
461 ABS(this_gap - r->rel_gap));
462 r->res_pri = this_pr;
463 r->res_dual = this_dr;
464 r->rel_gap = this_gap;
465 r->error_ratio =
466 MAX(r->res_pri / self->stgs->eps_p,
467 MAX(r->res_dual / self->stgs->eps_d, r->rel_gap / self->stgs->eps_g));
468
469 if (ABIP(dot)(self->c, &w->u[p], q) < 0) {
470 abip_float *Ax = (abip_float *)abip_malloc(p * sizeof(abip_float));
471 memset(Ax, 0, p * sizeof(abip_float));
472 self->spe_A_times(self, &w->u[p], Ax);
473 r->res_unbdd = ABIP(norm)(Ax, p) / (-ABIP(dot)(self->c, &w->u[p], q));
474 abip_free(Ax);
475 } else {
476 r->res_unbdd = INFINITY;
477 }
478
479 if (ABIP(dot)(self->b, w->u, p) > 0) {
480 abip_float *ATy_s = (abip_float *)abip_malloc(q * sizeof(abip_float));
481 memset(ATy_s, 0, q * sizeof(abip_float));
482 self->spe_AT_times(self, w->u, ATy_s);
483 ABIP(add_scaled_array)(ATy_s, &w->v_origin[p], q, 1);
484
485 r->res_infeas = ABIP(norm)(ATy_s, q) / ABIP(dot)(self->b, w->u, p);
486 abip_free(ATy_s);
487 } else {
488 r->res_infeas = INFINITY;
489 }
490
491 abip_free(x);
492 abip_free(beta_plus);
493 abip_free(beta_minus);
494 abip_free(z);
495 abip_free(s1);
496 abip_free(s2);
497 abip_free(pr);
498 abip_free(dr);
499 abip_free(dr1);
500 abip_free(dr2);
501 abip_free(lambda_ones);
502}

507cs *form_lasso_kkt(lasso *self) {
508 abip_int n = self->n;
509 abip_int m = self->m;
510 cs *LTL;
511
512 cs *Y1 = cs_spalloc(m, n, self->A->p[n], 1, 0);
513 memcpy(Y1->i, self->A->i, self->A->p[n] * sizeof(abip_int));
514 memcpy(Y1->p, self->A->p, (n + 1) * sizeof(abip_int));
515 memcpy(Y1->x, self->A->x, self->A->p[n] * sizeof(abip_float));
516
517 if (m > n) {
518 cs *Y2 = cs_spalloc(m, n, self->A->p[n], 1, 0);
519 memcpy(Y2->i, self->A->i, self->A->p[n] * sizeof(abip_int));
520 memcpy(Y2->p, self->A->p, (n + 1) * sizeof(abip_int));
521 memcpy(Y2->x, self->A->x, self->A->p[n] * sizeof(abip_float));
522
523 cs *T1 = cs_spalloc(n, n, n, 1, 1); /* create triplet identity matrix */
524
525 for (int i = 0; i < n; i++) {
526 cs_entry(T1, i, i, 0.5);
527 }
528
529 cs *half_eye = cs_compress(T1);
530
531 cs_spfree(T1);
532
533 for (int i = 0; i < Y1->p[n]; i++) {
534 Y1->x[i] /= self->D_hat[Y1->i[i]];
535 }
536
537 LTL = cs_add(half_eye, cs_multiply(cs_transpose(Y2, 1), Y1), 1, 1);
538
539 cs_spfree(Y1);
540 cs_spfree(Y2);
541 cs_spfree(half_eye);
542 } else {
543 cs *YYT = cs_multiply(Y1, cs_transpose(Y1, 1));
544 cs_spfree(Y1);
545
546 cs *diag = cs_spalloc(m, m, m, 1, 1);
547
548 for (int i = 0; i < m; i++) {
549 cs_entry(diag, i, i, self->D_hat[i]);
550 }
551 cs *diag_D_hat = cs_compress(diag);
552 cs_spfree(diag);
553
554 LTL = cs_add(diag_D_hat, YYT, 1, 2);
555 cs_spfree(YYT);
556 }
557
558 for (int i = 0; i < LTL->n; i++) {
559 for (int j = LTL->p[i]; j < LTL->p[i + 1]; j++) {
560 if (LTL->i[j] > i) LTL->x[j] = 0;
561 }
562 }
563 cs_dropzeros(LTL);
564 return LTL;
565}

571void init_lasso_precon(lasso *self) {
572 self->L->M = (abip_float *)abip_malloc(self->p * sizeof(abip_float));
573 memset(self->L->M, 0, self->p * sizeof(abip_float));
574
575 abip_float *M = self->L->M;
576
577 M[0] = 1 / (self->stgs->rho_y + 1);
578
579 for (int i = 0; i < self->A->p[self->A->n]; i++) {
580 M[self->A->i[i] + 1] += 2 * POWF(self->A->x[i], 2);
581 }
582
583 for (int i = 1; i < self->p; i++) {
584 M[i] = 1 / (self->stgs->rho_y + M[i] +
585 self->sc_cone2 * POWF(self->D[i - 1], 2));
586 }
587}

592abip_float get_lasso_pcg_tol(abip_int k, abip_float error_ratio,
593 abip_float norm_p) {
594 if (k == -1) {
595 return 1e-9 * norm_p;
596 } else {
597 if (error_ratio > 100000) {
598 return MAX(1e-9, 1.2e-2 * norm_p / POWF((k + 1), 2));
599 } else if (error_ratio > 30000) {
600 return MAX(1e-9, 8e-3 * norm_p / POWF((k + 1), 2));
601 } else if (error_ratio > 10000) {
602 return MAX(1e-9, 6e-3 * norm_p / POWF((k + 1), 2));
603 } else if (error_ratio > 3000) {
604 return MAX(1e-9, 5e-3 * norm_p / POWF((k + 1), 2));
605 } else if (error_ratio > 1000) {
606 return MAX(1e-9, 3e-3 * norm_p / POWF((k + 1), 2));
607 } else if (error_ratio > 300) {
608 return MAX(1e-9, 2e-3 * norm_p / POWF((k + 1), 2));
609 } else if (error_ratio > 100) {
610 return MAX(1e-9, 1.5e-3 * norm_p / POWF((k + 1), 2));
611 } else if (error_ratio > 30) {
612 return MAX(1e-9, 8e-4 * norm_p / POWF((k + 1), 2));
613 } else if (error_ratio > 10) {
614 return MAX(1e-9, 6e-4 * norm_p / POWF((k + 1), 2));
615 } else {
616 return MAX(1e-9, 5e-4 * norm_p / POWF((k + 1), 2));
617 }
618 }

648abip_int solve_lasso_linsys(lasso *self, abip_float *b,
649 abip_float *pcg_warm_start, abip_int iter,
650 abip_float error_ratio) {
651 ABIP(timer) linsys_timer;
652 ABIP(tic)(&linsys_timer);
653
654 ABIP(scale_array)(&b[self->p], -1, self->q);
655
656 if (self->stgs->linsys_solver == 3) { // pcg
657
658 abip_int p = self->p;
659 abip_float norm_p = ABIP(norm)(b, p);
660
661 self->spe_A_times(self, &b[p], b);
662 abip_float pcg_tol = get_lasso_pcg_tol(iter, error_ratio, norm_p);
663 abip_int cg_its = ABIP(solve_linsys)(self, b, p, pcg_warm_start, pcg_tol);
664
665 if (iter >= 0) {
666 self->L->total_cg_iters += cg_its;
667 }
668 }
669
670 else { // direct methods
671 abip_float *b2 = (abip_float *)abip_malloc(self->p * sizeof(abip_float));
672 memcpy(b2, b, self->p * sizeof(abip_float));
673 self->spe_A_times(self, &b[self->p], b2);
674 b[0] = b2[0] / (1 + self->stgs->rho_y);
675 abip_int n = self->n;
676 abip_int m = self->m;
677
678 if (m > n) {
679 for (int i = 0; i < m; i++) {
680 b2[i + 1] /= self->D_hat[i];
681 b[i + 1] = b2[i + 1];
682 }
683
684 abip_float *tmp = (abip_float *)abip_malloc(n * sizeof(abip_float));
685 memset(tmp, 0, n * sizeof(abip_float));
686 ABIP(accum_by_Atrans)(self->A, &b2[1], tmp);
687
688 ABIP(solve_linsys)(self, tmp, n, ABIP_NULL, 0);
689
690 abip_float *tmp2 = (abip_float *)abip_malloc(m * sizeof(abip_float));
691 memset(tmp2, 0, m * sizeof(abip_float));
692
693 ABIP(accum_by_A)(self->A, tmp, tmp2);
694 for (int i = 0; i < m; i++) {
695 tmp2[i] /= self->D_hat[i];
696 }
697
698 ABIP(add_scaled_array)(&b[1], tmp2, m, -1);
699
700 abip_free(tmp);
701 abip_free(tmp2);
702 } else {
703 memcpy(&b[1], &b2[1], m * sizeof(abip_float));
704
705 ABIP(solve_linsys)(self, &b[1], m, ABIP_NULL, 0);
706 }
707
708 abip_free(b2);
709 }
710
711 ABIP(scale_array)(&b[self->p], -1, self->q);
712 self->spe_AT_times(self, b, &b[self->p]);
713
714 self->L->total_solve_time += ABIP(tocq)(&linsys_timer);
715
716 return 0;
717}
