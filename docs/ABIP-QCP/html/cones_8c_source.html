128 K = {(t,x)| t>||x||}

130void ABIP(soc_barrier_subproblem)(abip_float* x, abip_float* tmp,
131 abip_float lambda, abip_int n) {
132 abip_float a = tmp[0];
133 abip_float tol = 1e-9;
134 abip_float* b = (abip_float*)abip_malloc((n - 1) * sizeof(abip_float));
135 memcpy(b, &tmp[1], (n - 1) * sizeof(abip_float));
136 abip_float b_norm_sq = ABIP(norm_sq)(b, n - 1);
137 if (ABS(a) <= tol) {
138 x[0] = SQRTF(2 * lambda + b_norm_sq / 4);
139 memcpy(&x[1], b, (n - 1) * sizeof(abip_float));
140 ABIP(scale_array)(&x[1], 0.5, (n - 1));
141 } else {
142 abip_float coef1 = -(POWF(a, 2) - ABIP(norm_sq)(b, n - 1)) / lambda;
143 abip_float coef2 =
144 -(4 * POWF(a, 2) + 4 * ABIP(norm_sq)(b, n - 1) + 16 * lambda) / lambda;
145
146 abip_float r = 16 * a * a /
147 (8 * lambda - a * a + b_norm_sq +
148 SQRTF(POWF((8 * lambda - a * a + b_norm_sq), 2) +
149 32 * a * a * lambda));
150 abip_float s1 = (r - SQRTF(r * (r + 8))) / 2;
151 abip_float s2 = (r + SQRTF(r * (r + 8))) / 2;
152
153 abip_float s = a > 0 ? s2 : s1;
154 abip_float eta = (s + 2) * a / s;
155 ABIP(scale_array)(b, (s + 2) / (s + 4), n - 1);
156
157 x[0] = eta;
158 memcpy(&x[1], b, (n - 1) * sizeof(abip_float));
159 }
160 abip_free(b);
161}

169void ABIP(rsoc_barrier_subproblem)(abip_float* x, abip_float* tmp,
170 abip_float lambda, abip_int n) {
171 abip_int nx = n - 2;
172 abip_float tol = 1e-9;
173
174 abip_float zeta_eta = tmp[0];
175 abip_float zeta_nu = tmp[1];
176
177 abip_float* zeta_x = (abip_float*)abip_malloc(nx * sizeof(abip_float));
178 memcpy(zeta_x, &tmp[2], nx * sizeof(abip_float));
179 abip_float zeta_x_norm_sq = ABIP(norm_sq)(zeta_x, nx);
180
181 if (zeta_eta + zeta_nu == 0) {
182 x[1] =
183 (-zeta_eta + SQRTF(zeta_eta * zeta_eta + 4 * lambda + zeta_x_norm_sq)) /
184 2;
185 x[0] = x[0] + zeta_eta;
186 memcpy(&x[2], zeta_x, nx * sizeof(abip_float));
187 ABIP(scale_array)(&x[2], 0.5, nx);
188 } else {
189 abip_float w;
190 abip_float s;
191 if (2 * zeta_eta * zeta_nu - zeta_x_norm_sq < 0) {
192 w = (2 * POWF(zeta_eta + zeta_nu, 2) / lambda) /
193 (-(2 * zeta_eta * zeta_nu - zeta_x_norm_sq) / (2 * lambda)) /
194 (1 + 4 / (-(2 * zeta_eta * zeta_nu - zeta_x_norm_sq) / (2 * lambda)) +
195 SQRTF(
196 1 +
197 (4 * (zeta_eta * zeta_eta + zeta_nu * zeta_nu + zeta_x_norm_sq) /
198 lambda +
199 16) /
200 (-(2 * zeta_eta * zeta_nu - zeta_x_norm_sq) / (2 * lambda)) /
201 (-(2 * zeta_eta * zeta_nu - zeta_x_norm_sq) /
202 (2 * lambda))));
203 } else {
204 w = (2 * zeta_eta * zeta_nu - zeta_x_norm_sq) / (2 * lambda) *
205 (1 - 4 / ((2 * zeta_eta * zeta_nu - zeta_x_norm_sq) / (2 * lambda)) +
206 SQRTF(
207 1 +
208 (4 * (zeta_eta * zeta_eta + zeta_nu * zeta_nu + zeta_x_norm_sq) /
209 lambda +
210 16) /
211 ((2 * zeta_eta * zeta_nu - zeta_x_norm_sq) / (2 * lambda)) /
212 ((2 * zeta_eta * zeta_nu - zeta_x_norm_sq) /
213 (2 * lambda)))) /
214 2;
215 }
216 if (zeta_eta + zeta_nu > 0) {
217 s = (w + SQRTF(w * (w + 4))) / 2;
218 x[0] = (zeta_eta * POWF(s + 1, 2) + zeta_nu * (s + 1)) / (s * (s + 2));
219 x[1] = (zeta_nu * POWF(s + 1, 2) + zeta_eta * (s + 1)) / (s * (s + 2));
220 memcpy(&x[2], zeta_x, nx * sizeof(abip_float));
221 ABIP(scale_array)(&x[2], (s + 1) / (s + 2), nx);
222 } else {
223 if (w > 10) {
224 s = 2 / (w + 2 + SQRTF(w * (w + 4)));
225 x[0] = (zeta_eta * POWF(s, 2) + zeta_nu * s) / ((s - 1) * (s + 1));
226 x[1] = (zeta_nu * POWF(s, 2) + zeta_eta * s) / ((s - 1) * (s + 1));
227 memcpy(&x[2], zeta_x, nx * sizeof(abip_float));
228 ABIP(scale_array)(&x[2], s / (s + 1), nx);
229 } else {
230 s = (w - SQRTF(w * (w + 4))) / 2;
231 x[0] = (zeta_eta * POWF(s + 1, 2) + zeta_nu * (s + 1)) / (s * (s + 2));
232 x[1] = (zeta_nu * POWF(s + 1, 2) + zeta_eta * (s + 1)) / (s * (s + 2));
233 memcpy(&x[2], zeta_x, nx * sizeof(abip_float));
234 ABIP(scale_array)(&x[2], (s + 1) / (s + 2), nx);
235 }
236 }
237 }
238
239 for (int i = nx + 2; i < n; i++) {
240 if (tmp[i] >= 0) {
241 x[i] = (tmp[i] + SQRTF(tmp[i] * tmp[i] + 4 * lambda)) / 2;
242 } else {
243 x[i] = 2 * lambda /
244 (-tmp[i] * (1 + SQRTF(1 + 4 * lambda / POWF(tmp[i], 2))));
245 }
246 }
247 abip_free(zeta_x);
248}
