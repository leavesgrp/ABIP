11/* AMD_aat: compute the symmetry of the pattern of A, and count the number of
12 * nonzeros each column of A+A' (excluding the diagonal). Assumes the input
13 * matrix has no errors, with sorted columns and no duplicates
14 * (AMD_valid (n, n, Ap, Ai) must be AMD_OK, but this condition is not
15 * checked).
16 */
17
18#include "amd_internal.h"
19
20GLOBAL size_t AMD_aat /* returns nz in A+A' */
21(
22 Int n,
23 const Int Ap [ ],
24 const Int Ai [ ],
25 Int Len [ ], /* Len [j]: length of column j of A+A', excl diagonal*/
26 Int Tp [ ], /* workspace of size n */
27 abip_float ABIPInfo [ ]
28)
29{
30 Int p1;
31 Int p2;
32 Int p;
33 Int pj;
34 Int pj2;
35
36 Int i;
37 Int j;
38 Int k;
39
40 Int nzdiag;
41 Int nzboth;
42 Int nz;
43
44 abip_float sym ;
45 size_t nzaat ;
46
47 #ifndef NDEBUG
48
49 AMD_debug_init ("AMD AAT") ;
50 for (k = 0 ; k < n ; k++) Tp [k] = EMPTY ;
51 ASSERT (AMD_valid (n, n, Ap, Ai) == AMD_OK) ;
52
53 #endif
54
55 if (ABIPInfo != (abip_float *) ABIP_NULL)
56 {
57 /* clear the ABIPInfo array, if it exists */
58 for (i = 0 ; i < AMD_INFO ; i++)
59 {
60 ABIPInfo [i] = EMPTY ;
61 }
62 ABIPInfo [AMD_STATUS] = AMD_OK ;
63 }
64
65 for (k = 0 ; k < n ; k++)
66 {
67 Len [k] = 0 ;
68 }
69
70 nzdiag = 0 ;
71 nzboth = 0 ;
72 nz = Ap [n] ;
73
74 for (k = 0 ; k < n ; k++)
75 {
76 p1 = Ap [k] ;
77 p2 = Ap [k+1] ;
78 AMD_DEBUG2 (("\nAAT Column: "ID" p1: "ID" p2: "ID"\n", k, p1, p2)) ;
79
80 /* construct A+A' */
81 for (p = p1 ; p < p2 ; )
82 {
83 /* scan the upper triangular part of A */
84 j = Ai [p] ;
85
86 if (j < k)
87 {
88 /* entry A (j,k) is in the strictly upper triangular part,
89 * add both A (j,k) and A (k,j) to the matrix A+A' */
90 Len [j]++ ;
91 Len [k]++ ;
92 AMD_DEBUG3 ((" upper ("ID","ID") ("ID","ID")\n", j,k, k,j));
93 p++ ;
94 }
95 else if (j == k)
96 {
97 /* skip the diagonal */
98 p++ ;
99 nzdiag++ ;
100 break ;
101 }
102 else /* j > k */
103 {
104 /* first entry below the diagonal */
105 break ;
106 }
107
108 /* scan lower triangular part of A, in column j until reaching
109 * row k. Start where last scan left off. */
110 ASSERT (Tp [j] != EMPTY) ;
111 ASSERT (Ap [j] <= Tp [j] && Tp [j] <= Ap [j+1]) ;
112 pj2 = Ap [j+1] ;
113
114 for (pj = Tp [j] ; pj < pj2 ; )
115 {
116 i = Ai [pj] ;
117
118 if (i < k)
119 {
120 /* A (i,j) is only in the lower part, not in upper.
121 * add both A (i,j) and A (j,i) to the matrix A+A' */
122 Len [i]++ ;
123 Len [j]++ ;
124 AMD_DEBUG3 ((" lower ("ID","ID") ("ID","ID")\n", i,j, j,i)) ;
125 pj++ ;
126 }
127 else if (i == k)
128 {
129 /* entry A (k,j) in lower part and A (j,k) in upper */
130 pj++ ;
131 nzboth++ ;
132 break ;
133 }
134 else /* i > k */
135 {
136 /* consider this entry later, when k advances to i */
137 break ;
138 }
139 }
140 Tp [j] = pj ;
141 }
142
143 /* Tp [k] points to the entry just below the diagonal in column k */
144 Tp [k] = p ;
145 }
146
147 /* clean up, for remaining mismatched entries */
148 for (j = 0 ; j < n ; j++)
149 {
150 for (pj = Tp [j] ; pj < Ap [j+1] ; pj++)
151 {
152 i = Ai [pj] ;
153
154 /* A (i,j) is only in the lower part, not in upper.
155 * add both A (i,j) and A (j,i) to the matrix A+A' */
156 Len [i]++ ;
157 Len [j]++ ;
158 AMD_DEBUG3 ((" lower cleanup ("ID","ID") ("ID","ID")\n", i,j, j,i)) ;
159 }
160 }
161
162 /* --------------------------------------------------------------------- */
163 /* compute the symmetry of the nonzero pattern of A */
164 /* --------------------------------------------------------------------- */
165
166 /* Given a matrix A, the symmetry of A is:
167 * B = tril (spones (A), -1) + triu (spones (A), 1) ;
168 * sym = nnz (B & B') / nnz (B) ;
169 * or 1 if nnz (B) is zero. */
170
171 if (nz == nzdiag)
172 {
173 sym = 1 ;
174 }
175 else
176 {
177 sym = (2 * (abip_float) nzboth) / ((abip_float) (nz - nzdiag)) ;
178 }
179
180 nzaat = 0 ;
181 for (k = 0 ; k < n ; k++)
182 {
183 nzaat += Len [k] ;
184 }
185
186 AMD_DEBUG1 (("AMD nz in A+A', excluding diagonal (nzaat) = %g\n", (abip_float) nzaat)) ;
187 AMD_DEBUG1 ((" nzboth: "ID" nz: "ID" nzdiag: "ID" symmetry: %g\n", nzboth, nz, nzdiag, sym)) ;
188
189 if (ABIPInfo != (abip_float *) ABIP_NULL)
190 {
191 ABIPInfo [AMD_STATUS] = AMD_OK ;
192 ABIPInfo [AMD_N] = n ;
193 ABIPInfo [AMD_NZ] = nz ;
194 ABIPInfo [AMD_SYMMETRY] = sym ; /* symmetry of pattern of A */
195 ABIPInfo [AMD_NZDIAG] = nzdiag ; /* nonzeros on diagonal of A */
196 ABIPInfo [AMD_NZ_A_PLUS_AT] = nzaat ; /* nonzeros in A+A' */
197 }
198
199 return (nzaat) ;
200}
