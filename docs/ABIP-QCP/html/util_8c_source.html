50abip_float ABIP(tocq)(ABIP(timer) * t) {
51 struct timespec temp;
52
53 clock_gettime(CLOCK_MONOTONIC, &t->toc);
54
55 if ((t->toc.tv_nsec - t->tic.tv_nsec) < 0) {
56 temp.tv_sec = t->toc.tv_sec - t->tic.tv_sec - 1;
57 temp.tv_nsec = 1e9 + t->toc.tv_nsec - t->tic.tv_nsec;
58 } else {
59 temp.tv_sec = t->toc.tv_sec - t->tic.tv_sec;
60 temp.tv_nsec = t->toc.tv_nsec - t->tic.tv_nsec;
61 }
62
63 return (abip_float)temp.tv_sec * 1e3 + (abip_float)temp.tv_nsec / 1e6;
64}
