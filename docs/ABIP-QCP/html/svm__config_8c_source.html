202void svm_AT_times(svm *self, const abip_float *x, abip_float *y) {
203 y[0] += x[0];
204 abip_int m = self->m;
205 abip_int n = self->n;
206
207 for (int i = 0; i < n; i++) {
208 y[i + 2] += self->wD[i] * x[i + m + 1];
209 }
210
211 abip_float *tmp = (abip_float *)abip_malloc(n * sizeof(abip_float));
212 for (int i = 0; i < n; i++) {
213 tmp[i] = -self->wE[i] * x[i + m + 1];
214 }
215 ABIP(accum_by_Atrans)(self->wA, &x[1], tmp);
216
217 ABIP(add_scaled_array)(&y[n + 2], tmp, n, 1);
218 ABIP(add_scaled_array)(&y[2 * n + 3], tmp, n, -1);
219
220 y[2 * n + 2] += ABIP(dot)(self->wy, &x[1], m);
221 y[3 * n + 3] -= ABIP(dot)(self->wy, &x[1], m);
222
223 for (int i = 0; i < m; i++) {
224 y[i + 3 * n + 4] += self->wB[i] * x[i + 1];
225 y[i + 3 * n + 4 + m] -= self->wC[i] * x[i + 1];
226 }
227
228 abip_free(tmp);

283void scaling_svm_data(svm *self, ABIPCone *k) {
284 if (!ABIP(copy_A_matrix)(&(self->A), self->data->A)) {
285 abip_printf("ERROR: copy A matrix failed\n");
286 RETURN;
287 }
288
289 abip_int m = self->m;
290 abip_int n = self->n + 1;
291 ABIPMatrix *A = self->A;
292
293 abip_float min_row_scale = MIN_SCALE * SQRTF((abip_float)n);
294 abip_float max_row_scale = MAX_SCALE * SQRTF((abip_float)n);
295 abip_float min_col_scale = MIN_SCALE * SQRTF((abip_float)m);
296 abip_float max_col_scale = MAX_SCALE * SQRTF((abip_float)m);
297
298 abip_float *E = self->sc_E;
299 memset(E, 0, n * sizeof(abip_float));
300
301 abip_float *D = self->sc_D;
302 memset(D, 0, m * sizeof(abip_float));
303
304 abip_float avg = 0;
305
306 if (self->stgs->scale_E) {
307 for (int i = 0; i < n; i++) {
308 for (int j = A->p[i]; j < A->p[i + 1]; j++) {
309 E[i] += A->x[j] * A->x[j];
310 }
311 E[i] = SQRTF(E[i]);
312 avg += E[i];
313 }
314
315 avg /= n;
316
317 for (int i = 0; i < n; i++) {
318 E[i] = avg / E[i];
319 }
320
321 for (int i = 0; i < n; i++) {
322 for (int j = A->p[i]; j < A->p[i + 1]; j++) {
323 A->x[j] *= E[i];
324 }
325 }
326 }
327
328 for (int i = 0; i < A->p[n]; i++) {
329 D[A->i[i]] += A->x[i] * A->x[i];
330 }
331
332 avg = 0;
333 for (int i = 0; i < m; i++) {
334 avg += SQRTF(D[i]);
335 }
336
337 avg /= m;
338
339 for (int i = 0; i < m; i++) {
340 D[i] = avg / SQRTF(D[i]);
341 }
342
343 for (int i = 0; i < A->p[n]; i++) {
344 A->x[i] *= D[A->i[i]];
345 }
346
347 for (int i = 0; i < n - 1; i++) {
348 self->sc_F[i] = 1 / SQRTF(1 + 2 * E[i] * E[i]);
349 }
350
351 memcpy(self->b, self->data->b, self->p * sizeof(abip_float));
352
353 self->b[0] = self->sc_cone2;
354
355 for (int i = 1; i < m + 1; i++) {
356 self->b[i] = D[i - 1];
357 }
358 ABIP(scale_array)(self->b, self->sc_b, self->p);
359
360 memcpy(self->c, self->data->c, self->q * sizeof(abip_float));
361 self->c[0] = 0;
362 self->c[1] = self->sc_c * self->sc_cone1 * self->sc_cone2;
363
364 for (int i = 0; i < m; i++) {
365 self->c[i + 4 + 3 * self->n] = self->lambda * self->sc_c / self->sc;
366 }
367
368 n -= 1;
369
370 ABIP(copy_A_matrix)(&(self->wA), self->A);
371 self->wA->n -= 1;
372
373 memcpy(self->wy, &self->A->x[self->A->p[n]], m * sizeof(abip_float));
374
375 memcpy(self->wB, self->sc_D, m * sizeof(abip_float));
376 ABIP(scale_array)(self->wB, 1 / self->sc, m);
377
378 memcpy(self->wC, self->sc_D, m * sizeof(abip_float));
379
380 memcpy(self->wD, self->sc_F, n * sizeof(abip_float));
381 ABIP(scale_array)(self->wD, -SQRTF(self->sc_cone1), n);
382
383 for (int i = 0; i < n; i++) {
384 self->wE[i] = self->sc_E[i] * self->sc_F[i];
385 }
386
387 for (int i = 0; i < m; i++) {
388 self->wF[i] = self->wB[i] * self->wB[i] + self->wC[i] * self->wC[i] +
389 self->stgs->rho_y;
390 }
391
392 for (int i = 0; i < n; i++) {
393 self->wG[i] = self->wD[i] * self->wD[i] + 2 * self->wE[i] * self->wE[i] +
394 self->stgs->rho_y;
395 }
396
397 for (int i = 0; i < n; i++) {
398 self->wH[i] = 2 - 4 / self->wG[i] * self->wE[i] * self->wE[i];
399 }
400 self->wH[n] = 2;
401
402 ABIP(copy_A_matrix)(&(self->wX), self->wA);
403 for (int i = 0; i < n; i++) {
404 for (int j = self->wX->p[i]; j < self->wX->p[i + 1]; j++) {
405 self->wX->x[j] *= (-2 * self->wE[i]);
406 }
407 }
408}

445void calc_svm_residuals(svm *self, ABIPWork *w, ABIPResiduals *r,
446 abip_int ipm_iter, abip_int admm_iter) {
447 abip_int p = w->m;
448 abip_int q = w->n;
449 abip_int m = self->m;
450 abip_int n = self->n;
451 abip_float C = self->lambda;
452 abip_float this_pr;
453 abip_float this_dr;
454 abip_float this_gap;
455
456 r->tau = w->u[p + q];
457
458 abip_float *w1 = (abip_float *)abip_malloc((n + 1) * sizeof(abip_float));
459 memcpy(w1, &w->u[m + 2 * n + 3], n * sizeof(abip_float));
460 ABIP(add_scaled_array)(w1, &w->u[m + 3 * n + 4], n, -1);
461 for (int i = 0; i < n; i++) {
462 w1[i] *= (self->sc_E[i] / (r->tau * self->sc_b));
463 }
464
465 abip_float b = (w->u[m + 3 * n + 3] - w->u[m + 4 * n + 4]) / r->tau *
466 self->sc_E[n] / self->sc_b;
467 w1[n] = b;
468
469 abip_float *xi = (abip_float *)abip_malloc(m * sizeof(abip_float));
470 memcpy(xi, &w->u[m + 4 * n + 5], m * sizeof(abip_float));
471 ABIP(scale_array)(xi, 1 / (r->tau * self->sc * self->sc_b), m);
472
473 abip_float *t = (abip_float *)abip_malloc(m * sizeof(abip_float));
474 memcpy(t, &w->u[2 * m + 4 * n + 5], m * sizeof(abip_float));
475 ABIP(scale_array)(t, 1 / (r->tau * self->sc_b), m);
476
477 abip_float *y = (abip_float *)abip_malloc(m * sizeof(abip_float));
478 memcpy(y, &w->u[1], m * sizeof(abip_float));
479 for (int i = 0; i < m; i++) {
480 y[i] = y[i] / r->tau * self->sc_D[i] / self->sc_c;
481 }
482
483 abip_float *s2 = (abip_float *)abip_malloc(m * sizeof(abip_float));
484 memcpy(s2, &w->v[2 * m + 4 * n + 5], m * sizeof(abip_float));
485 ABIP(scale_array)(s2, 1 / (r->tau * self->sc_c), m);
486
487 abip_float *s1 = (abip_float *)abip_malloc(m * sizeof(abip_float));
488 memcpy(s1, &w->v[m + 4 * n + 5], m * sizeof(abip_float));
489 ABIP(scale_array)(s1, self->sc / (r->tau * self->sc_c), m);
490
491 abip_float *pr = (abip_float *)abip_malloc(m * sizeof(abip_float));
492 memcpy(pr, xi, m * sizeof(abip_float));
493 ABIP(accum_by_A)(self->data->A, w1, pr);
494 for (int i = 0; i < m; i++) {
495 pr[i] -= (t[i] + 1);
496 }
497 this_pr = ABIP(norm)(pr, m) / SQRTF(m);
498
499 abip_float *dr = (abip_float *)abip_malloc(2 * m * sizeof(abip_float));
500 for (int i = 0; i < m; i++) {
501 dr[i] = y[i] - s2[i];
502 dr[i + m] = y[i] + s1[i] - C;
503 }
504 this_dr = ABIP(norm)(dr, 2 * m) / (SQRTF(m) * C);
505
506 ABIPMatrix *B = (ABIPMatrix *)abip_malloc(sizeof(ABIPMatrix));
507 ABIP(copy_A_matrix)(&B, self->data->A);
508 B->n -= 1;
509 abip_float *BTy = (abip_float *)abip_malloc(n * sizeof(abip_float));
510 memset(BTy, 0, n * sizeof(abip_float));
511 ABIP(accum_by_Atrans)(B, y, BTy);
512
513 r->dobj = 0;
514 r->pobj = 0;
515 for (int i = 0; i < m; i++) {
516 r->dobj += y[i];
517 r->pobj += C * xi[i];
518 }
519
520 r->pobj += 0.5 * ABIP(dot)(w1, w1, n);
521 r->dobj -= 0.5 * ABIP(dot)(BTy, BTy, n);
522
523 this_gap = r->dobj - r->pobj;
524 this_gap = ABS(this_gap) / (1 + ABS(r->pobj));
525
526 r->last_ipm_iter = ipm_iter;
527 r->last_admm_iter = admm_iter;
528
529 r->res_infeas = NAN;
530 r->res_unbdd = NAN;
531
532 r->res_dif = MAX(MAX(ABS(this_pr - r->res_pri), ABS(this_dr - r->res_dual)),
533 ABS(this_gap - r->rel_gap));
534 r->res_pri = this_pr;
535 r->res_dual = this_dr;
536 r->rel_gap = this_gap;
537 r->error_ratio =
538 MAX(r->res_pri / self->stgs->eps_p,
539 MAX(r->res_dual / self->stgs->eps_d, r->rel_gap / self->stgs->eps_g));
540
541 if (ABIP(dot)(self->c, &w->u[p], q) < 0) {
542 abip_float *Ax = (abip_float *)abip_malloc(p * sizeof(abip_float));
543 memset(Ax, 0, p * sizeof(abip_float));
544 self->spe_A_times(self, &w->u[p], Ax);
545 r->res_unbdd = ABIP(norm)(Ax, p) / (-ABIP(dot)(self->c, &w->u[p], q));
546 abip_free(Ax);
547 } else {
548 r->res_unbdd = INFINITY;
549 }
550
551 if (ABIP(dot)(self->b, w->u, p) > 0) {
552 abip_float *ATy_s = (abip_float *)abip_malloc(q * sizeof(abip_float));
553 memset(ATy_s, 0, q * sizeof(abip_float));
554 self->spe_AT_times(self, w->u, ATy_s);
555 ABIP(add_scaled_array)(ATy_s, &w->v_origin[p], q, 1);
556
557 r->res_infeas = ABIP(norm)(ATy_s, q) / ABIP(dot)(self->b, w->u, p);
558 abip_free(ATy_s);
559 } else {
560 r->res_infeas = INFINITY;
561 }
562
563 ABIP(free_A_matrix)(B);
564 abip_free(w1);
565 abip_free(xi);
566 abip_free(t);
567 abip_free(y);
568 abip_free(s1);
569 abip_free(s2);
570 abip_free(pr);
571 abip_free(dr);

577cs *form_svm_kkt(svm *self) {
578 abip_int n = self->n;
579 abip_int m = self->m;
580 cs *LTL;
581
582 cs *N1 = cs_spalloc(m, n + 1, self->A->p[n + 1], 1, 0);
583 memcpy(N1->i, self->A->i, self->A->p[n + 1] * sizeof(abip_int));
584 memcpy(N1->p, self->A->p, (n + 2) * sizeof(abip_int));
585 memcpy(N1->x, self->A->x, self->A->p[n + 1] * sizeof(abip_float));
586
587 cs *N2 = cs_spalloc(m, n + 1, self->A->p[n + 1], 1, 0);
588 memcpy(N2->i, self->A->i, self->A->p[n + 1] * sizeof(abip_int));
589 memcpy(N2->p, self->A->p, (n + 2) * sizeof(abip_int));
590 memcpy(N2->x, self->A->x, self->A->p[n + 1] * sizeof(abip_float));
591
592 if (m > n + 1) {
593 for (int i = 0; i < N1->p[n + 1]; i++) {
594 N1->x[i] /= self->wF[N1->i[i]];
595 }
596
597 cs *diag = cs_spalloc(n + 1, n + 1, n + 1, 1, 1);
598
599 for (int i = 0; i < n + 1; i++) {
600 cs_entry(diag, i, i, 1 / self->wH[i]);
601 }
602 cs *diag_H = cs_compress(diag);
603 cs_spfree(diag);
604
605 LTL = cs_add(diag_H, cs_multiply(cs_transpose(N2, 1), N1), 1, 1);
606 cs_spfree(diag_H);
607 } else {
608 for (int i = 0; i < N1->n; i++) {
609 for (int j = N1->p[i]; j < N1->p[i + 1]; j++) {
610 N1->x[j] *= self->wH[i];
611 }
612 }
613 cs *diag = cs_spalloc(m, m, m, 1, 1);
614
615 for (int i = 0; i < m; i++) {
616 cs_entry(diag, i, i, self->wF[i]);
617 }
618 cs *diag_F = cs_compress(diag);
619 cs_spfree(diag);
620
621 LTL = cs_add(diag_F, cs_multiply(N1, cs_transpose(N2, 1)), 1, 1);
622
623 cs_spfree(diag_F);
624 }
625
626 cs_spfree(N1);
627 cs_spfree(N2);
628
629 for (int i = 0; i < LTL->n; i++) {
630 for (int j = LTL->p[i]; j < LTL->p[i + 1]; j++) {
631 if (LTL->i[j] > i) LTL->x[j] = 0;
632 }
633 }
634 cs_dropzeros(LTL);
635 return LTL;
636}

642void init_svm_precon(svm *self) {
643 self->L->M = (abip_float *)abip_malloc(self->p * sizeof(abip_float));
644 memset(self->L->M, 0, self->p * sizeof(abip_float));
645
646 abip_float *M = self->L->M;
647
648 M[0] = 1 / (self->stgs->rho_y + 1);

660 M[i] = 1 / (self->stgs->rho_y + 2 * POWF(self->wE[i - self->m + 1], 2) +
661 POWF(self->wD[i - self->m + 1], 2));

669abip_float get_svm_pcg_tol(abip_int k, abip_float error_ratio,
670 abip_float norm_p) {
671 if (k == -1) {
672 return 1e-9 * norm_p;
673 } else {
674 if (error_ratio > 100000) {
675 return MAX(1e-9, 3e-2 * norm_p / POWF((k + 1), 2));
676 } else if (error_ratio > 30000) {
677 return MAX(1e-9, 3e-2 * norm_p / POWF((k + 1), 2));
678 } else if (error_ratio > 10000) {
679 return MAX(1e-9, 3e-2 * norm_p / POWF((k + 1), 2));
680 } else if (error_ratio > 3000) {
681 return MAX(1e-9, 2.5e-2 * norm_p / POWF((k + 1), 2));
682 } else if (error_ratio > 1000) {
683 return MAX(1e-9, 2e-2 * norm_p / POWF((k + 1), 2));
684 } else if (error_ratio > 300) {
685 return MAX(1e-9, 1.6e-2 * norm_p / POWF((k + 1), 2));
686 } else if (error_ratio > 100) {
687 return MAX(1e-9, 1.3e-2 * norm_p / POWF((k + 1), 2));
688 } else if (error_ratio > 30) {
689 return MAX(1e-9, 1e-2 * norm_p / POWF((k + 1), 2));
690 } else if (error_ratio > 10) {
691 return MAX(1e-9, 7e-3 * norm_p / POWF((k + 1), 2));
692 } else {
693 return MAX(1e-9, 4e-3 * norm_p / POWF((k + 1), 2));
694 }
695 }

702abip_int init_svm_linsys_work(svm *self) {
703 if (self->stgs->linsys_solver == 0) { // mkl-dss need lower triangle
704 self->L->K = cs_transpose(form_svm_kkt(self), 1);
705 } else if (self->stgs->linsys_solver == 1) { // qdldl need upper triangle
706 self->L->K = form_svm_kkt(self);
707 } else if (self->stgs->linsys_solver == 2) { // cholesky need upper triangle
708 self->L->K = form_svm_kkt(self);
709 } else if (self->stgs->linsys_solver == 3) { // pcg doesn't need kkt matrix
710 init_svm_precon(self);
711 self->L->K = ABIP_NULL;
712 } else if (self->stgs->linsys_solver ==
713 4) { // mkl-pardiso need lower triangle
714 self->L->K = cs_transpose(form_svm_kkt(self), 1);
715 } else if (self->stgs->linsys_solver ==
716 5) { // dense cholesky need upper triangle
717 self->L->K = form_svm_kkt(self);
718 } else {
719 printf("\nlinsys solver type error\n");
720 return -1;
721 }
722 return ABIP(init_linsys_work)(self);

728abip_int solve_svm_linsys(svm *self, abip_float *b, abip_float *pcg_warm_start,
729 abip_int iter, abip_float error_ratio) {
730 ABIP(timer) linsys_timer;
731 ABIP(tic)(&linsys_timer);
732
733 ABIP(scale_array)(&b[self->p], -1, self->q);
734
735 if (self->stgs->linsys_solver == 3) { // pcg
736
737 abip_int p = self->p;
738 abip_float norm_p = ABIP(norm)(b, p);
739
740 self->spe_A_times(self, &b[p], b);
741 abip_float pcg_tol = get_svm_pcg_tol(iter, error_ratio, norm_p);
742 abip_int cg_its = ABIP(solve_linsys)(self, b, p, pcg_warm_start, pcg_tol);
743
744 if (iter >= 0) {
745 self->L->total_cg_iters += cg_its;
746 }
747 } else { // direct methods
748 abip_int n = self->n;
749 abip_int m = self->m;
750 abip_float *b2 = (abip_float *)abip_malloc(self->p * sizeof(abip_float));
751 memcpy(b2, b, self->p * sizeof(abip_float));
752 self->spe_A_times(self, &b[self->p], b2);
753 b[0] = b2[0] / (1 + self->stgs->rho_y);
754
755 for (int i = 0; i < n; i++) {
756 b[i + m + 1] = b2[i + m + 1] / self->wG[i];
757 }
758
759 abip_float *b3 = (abip_float *)abip_malloc(m * sizeof(abip_float));
760 memcpy(b3, &b2[1], m * sizeof(abip_float));
761 ABIP(scale_array)(b3, -1, m);
762 ABIP(accum_by_A)(self->wX, &b[m + 1], b3);
763 ABIP(scale_array)(b3, -1, m);
764
765 abip_float *tmp = (abip_float *)abip_malloc((n + 1) * sizeof(abip_float));
766 if (m > n + 1) {
767 for (int i = 0; i < m; i++) {
768 b3[i] /= self->wF[i];
769 }
770 memset(tmp, 0, (n + 1) * sizeof(abip_float));
771 ABIP(accum_by_Atrans)(self->A, b3, tmp);
772 ABIP(solve_linsys)(self, tmp, n + 1, ABIP_NULL, 0);
773
774 abip_float *tmp2 = (abip_float *)abip_malloc(m * sizeof(abip_float));
775 memset(tmp2, 0, m * sizeof(abip_float));
776
777 ABIP(accum_by_A)(self->A, tmp, tmp2);
778 for (int i = 0; i < m; i++) {
779 tmp2[i] /= self->wF[i];
780 }
781 memcpy(&b[1], b3, m * sizeof(abip_float));
782 ABIP(add_scaled_array)(&b[1], tmp2, m, -1);
783 abip_free(tmp2);
784 } else {
785 ABIP(solve_linsys)(self, b3, m, ABIP_NULL, 0);
786 memcpy(&b[1], b3, m * sizeof(abip_float));
787 }
788
789 memset(tmp, 0, (n + 1) * sizeof(abip_float));
790 ABIP(accum_by_Atrans)(self->wX, &b[1], tmp);
791 for (int i = 0; i < n; i++) {
792 tmp[i] /= self->wG[i];
793 }
794 ABIP(add_scaled_array)(&b[m + 1], tmp, n, -1);
795
796 abip_free(b2);
797 abip_free(b3);
798 abip_free(tmp);
799 }
800
801 ABIP(scale_array)(&b[self->p], -1, self->q);
802 self->spe_AT_times(self, b, &b[self->p]);
803
804 self->L->total_solve_time += ABIP(tocq)(&linsys_timer);
805
806 return 0;
807}
