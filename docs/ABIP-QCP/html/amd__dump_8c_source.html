54/* Dump AMD's data structure, except for the hash buckets. This routine
55 * cannot be called when the hash buckets are non-empty.
56 */
57
58GLOBAL void AMD_dump (
59 Int n, /* A is n-by-n */
60 Int Pe [ ], /* pe [0..n-1]: index in iw of start of row i */
61 Int Iw [ ], /* workspace of size iwlen, iwlen [0..pfree-1]
62 * holds the matrix on input */
63 Int Len [ ], /* len [0..n-1]: length for row i */
64 Int iwlen, /* length of iw */
65 Int pfree, /* iw [pfree ... iwlen-1] is empty on input */
66 Int Nv [ ], /* nv [0..n-1] */
67 Int Next [ ], /* next [0..n-1] */
68 Int Last [ ], /* last [0..n-1] */
69 Int Head [ ], /* head [0..n-1] */
70 Int Elen [ ], /* size n */
71 Int Degree [ ], /* size n */
72 Int W [ ], /* size n */
73 Int nel
74)
75{
76 Int i;
77 Int pe;
78 Int elen;
79 Int nv;
80 Int len;
81 Int e;
82 Int p;
83 Int k;
84 Int j;
85 Int deg;
86 Int w;
87 Int cnt;
88 Int ilast;
89
90 if (AMD_debug < 0) return ;
91 ASSERT (pfree <= iwlen) ;
92 AMD_DEBUG3 (("\nAMD dump, pfree: "ID"\n", pfree)) ;
93 for (i = 0 ; i < n ; i++)
94 {
95 pe = Pe [i] ;
96 elen = Elen [i] ;
97 nv = Nv [i] ;
98 len = Len [i] ;
99 w = W [i] ;
100
101 if (elen >= EMPTY)
102 {
103 if (nv == 0)
104 {
105 AMD_DEBUG3 (("\nI "ID": nonprincipal: ", i)) ;
106 ASSERT (elen == EMPTY) ;
107
108 if (pe == EMPTY)
109 {
110 AMD_DEBUG3 ((" dense node\n")) ;
111 ASSERT (w == 1) ;
112 }
113 else
114 {
115 ASSERT (pe < EMPTY) ;
116 AMD_DEBUG3 ((" i "ID" -> parent "ID"\n", i, FLIP (Pe[i])));
117 }
118 }
119 else
120 {
121 AMD_DEBUG3 (("\nI "ID": active principal supervariable:\n",i));
122 AMD_DEBUG3 ((" nv(i): "ID" Flag: %d\n", nv, (nv < 0))) ;
123
124 ASSERT (elen >= 0) ;
125 ASSERT (nv > 0 && pe >= 0) ;
126 p = pe ;
127 AMD_DEBUG3 ((" e/s: ")) ;
128
129 if (elen == 0) AMD_DEBUG3 ((" : ")) ;
130 ASSERT (pe + len <= pfree) ;
131
132 for (k = 0 ; k < len ; k++)
133 {
134 j = Iw [p] ;
135 AMD_DEBUG3 ((" "ID"", j)) ;
136 ASSERT (j >= 0 && j < n) ;
137
138 if (k == elen-1) AMD_DEBUG3 ((" : ")) ;
139 p++ ;
140 }
141
142 AMD_DEBUG3 (("\n")) ;
143 }
144 }
145 else
146 {
147 e = i ;
148
149 if (w == 0)
150 {
151 AMD_DEBUG3 (("\nE "ID": absorbed element: w "ID"\n", e, w)) ;
152 ASSERT (nv > 0 && pe < 0) ;
153 AMD_DEBUG3 ((" e "ID" -> parent "ID"\n", e, FLIP (Pe [e]))) ;
154 }
155 else
156 {
157 AMD_DEBUG3 (("\nE "ID": unabsorbed element: w "ID"\n", e, w)) ;
158 ASSERT (nv > 0 && pe >= 0) ;
159 p = pe ;
160 AMD_DEBUG3 ((" : ")) ;
161 ASSERT (pe + len <= pfree) ;
162
163 for (k = 0 ; k < len ; k++)
164 {
165 j = Iw [p] ;
166 AMD_DEBUG3 ((" "ID"", j)) ;
167 ASSERT (j >= 0 && j < n) ;
168 p++ ;
169 }
170
171 AMD_DEBUG3 (("\n")) ;
172 }
173 }
174 }
175
176 /* this routine cannot be called when the hash buckets are non-empty */
177 AMD_DEBUG3 (("\nDegree lists:\n")) ;
178 if (nel >= 0)
179 {
180 cnt = 0 ;
181
182 for (deg = 0 ; deg < n ; deg++)
183 {
184 if (Head [deg] == EMPTY) continue ;
185 ilast = EMPTY ;
186 AMD_DEBUG3 ((ID": \n", deg)) ;
187
188 for (i = Head [deg] ; i != EMPTY ; i = Next [i])
189 {
190 AMD_DEBUG3 ((" "ID" : next "ID" last "ID" deg "ID"\n", i, Next [i], Last [i], Degree [i])) ;
191 ASSERT (i >= 0 && i < n && ilast == Last [i] && deg == Degree [i]) ;
192 cnt += Nv [i] ;
193 ilast = i ;
194 }
195
196 AMD_DEBUG3 (("\n")) ;
197 }
198
199 ASSERT (cnt == n - nel) ;
200 }
201}
202
203#endif
